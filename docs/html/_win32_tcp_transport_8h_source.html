190 int SendToSocket(SOCKET s, dmq::xostringstream& os, const DmqHeader& header) {
191 if (s == INVALID_SOCKET) return -1;
192
193 DmqHeader headerCopy = header;
194 auto payload = os.str();
195 if (payload.length() > UINT16_MAX) return -1;
196 headerCopy.SetLength(static_cast<uint16_t>(payload.length()));
197
198 dmq::xostringstream ss(std::ios::in | std::ios::out | std::ios::binary);
199
200 // Convert to Network Byte Order (Big Endian)
201 uint16_t marker = htons(headerCopy.GetMarker());
202 uint16_t id = htons(headerCopy.GetId());
203 uint16_t seq = htons(headerCopy.GetSeqNum());
204 uint16_t len = htons(headerCopy.GetLength());
205
206 ss.write((char*)&marker, 2);
207 ss.write((char*)&id, 2);
208 ss.write((char*)&seq, 2);
209 ss.write((char*)&len, 2);
210 ss.write(payload.data(), payload.size());
211
212 auto data = ss.str();
213 const char* ptr = data.c_str();
214 int remaining = (int)data.length();
215
216 while (remaining > 0)
217 {
218 int sent = send(s, ptr, remaining, 0);
219 if (sent == SOCKET_ERROR) return -1;
220 ptr += sent;
221 remaining -= sent;
222 }
223
224 // Always track the message (unless it is an ACK)
225 if (headerCopy.GetId() != dmq::ACK_REMOTE_ID && m_transportMonitor)
226 m_transportMonitor->Add(headerCopy.GetSeqNum(), headerCopy.GetId());
227
228 return 0;
229 }

232 int ReceiveServer(dmq::xstringstream& is, DmqHeader& header) {
233 // 1. Lazy Accept: Check for new client connections
234 TIMEVAL tv = { 0, 1000 }; // 1ms poll
235 fd_set fds;
236 FD_ZERO(&fds);
237 FD_SET(m_listenSocket, &fds);
238
239 if (select(0, &fds, NULL, NULL, &tv) > 0) {
240 SOCKET client = accept(m_listenSocket, NULL, NULL);
241 if (client != INVALID_SOCKET) {
242 std::cout << "Win32TcpTransport: Accepted client connection." << std::endl;
243 setsockopt(client, SOL_SOCKET, SO_RCVTIMEO, (const char*)&m_recvTimeoutMs, sizeof(m_recvTimeoutMs));
244 m_serverClients.push_back(client);
245 }
246 }
247
248 // 2. Poll all connected clients for data
249 if (m_serverClients.empty()) return -1;
250
251 fd_set readfds;
252 FD_ZERO(&readfds);
253 for (auto s : m_serverClients) {
254 FD_SET(s, &readfds);
255 }
256
257 tv.tv_usec = 1000; // 1ms poll
258 if (select(0, &readfds, NULL, NULL, &tv) > 0) {
259 for (auto it = m_serverClients.begin(); it != m_serverClients.end(); ) {
260 if (FD_ISSET(*it, &readfds)) {
261 int result = ReceiveSocket(*it, is, header);
262 if (result == -2) { // Disconnected
263 std::cout << "Win32TcpTransport: Client disconnected." << std::endl;
264 closesocket(*it);
265 it = m_serverClients.erase(it);
266 continue;
267 }
268 return result;
269 }
270 ++it;
271 }
272 }
273 return -1;

277 int ReceiveSocket(SOCKET s, dmq::xstringstream& is, DmqHeader& header) {
278 if (s == INVALID_SOCKET) return -1;
279
280 // Poll check to prevent blocking if no data is waiting
281 fd_set readfds;
282 FD_ZERO(&readfds);
283 FD_SET(s, &readfds);
284 TIMEVAL tv = { 0, 1000 };
285 if (select(0, &readfds, NULL, NULL, &tv) <= 0) return -1;
286
287 // 1. Read Header
288 char headerBuf[DmqHeader::HEADER_SIZE];
289 if (!ReadExact(s, headerBuf, DmqHeader::HEADER_SIZE)) return -2; // Disconnected
290
291 dmq::xstringstream hs(std::ios::in | std::ios::out | std::ios::binary);
292 hs.write(headerBuf, DmqHeader::HEADER_SIZE);
293 hs.seekg(0);
294
295 uint16_t val;
296
297 // Read Marker (Convert Network -> Host)
298 hs.read((char*)&val, 2); header.SetMarker(ntohs(val));
299 if (header.GetMarker() != DmqHeader::MARKER) return -1;
300
301 hs.read((char*)&val, 2); header.SetId(ntohs(val));
302 hs.read((char*)&val, 2); header.SetSeqNum(ntohs(val));
303 hs.read((char*)&val, 2); header.SetLength(ntohs(val));
304
305 // 2. Read Payload
306 uint16_t length = header.GetLength();
307 if (length > 0)
308 {
309 m_recvBuf.resize(length);
310 if (!ReadExact(s, m_recvBuf.data(), length)) return -2;
311 is.write(m_recvBuf.data(), length);
312 }
313
314 // 3. Handle Acknowledgment
315 if (header.GetId() == dmq::ACK_REMOTE_ID) {
316 if (m_transportMonitor) m_transportMonitor->Remove(header.GetSeqNum());
317 }
318 else if (m_transportMonitor && m_sendTransport) {
319 dmq::xostringstream ss_ack;
320 DmqHeader ack;
321 ack.SetId(dmq::ACK_REMOTE_ID);
322 ack.SetSeqNum(header.GetSeqNum());
323 m_sendTransport->Send(ss_ack, ack);
324 }
325 return 0;
326 }

329 bool ReadExact(SOCKET s, char* dest, int size)
330 {
331 int total = 0;
332 while (total < size) {
333 int r = recv(s, dest + total, size - total, 0);
334 if (r <= 0) return false;
335 total += r;
336 }
337 return true;
338 }
