61 int Create(Type type, LPCSTR addr, USHORT port)
62 {
63 m_type = type;
64 sockaddr_in service;
65 service.sin_family = AF_INET;
66 service.sin_port = htons(port);
67 inet_pton(AF_INET, addr, &service.sin_addr);
68
69 if (type == Type::SERVER)
70 {
71 m_listenSocket = socket(AF_INET, SOCK_STREAM, IPPROTO_TCP);

73
74 if (bind(m_listenSocket, (SOCKADDR*)&service, sizeof(service)) == SOCKET_ERROR)

132 virtual int Send(dmq::xostringstream& os, const DmqHeader& header) override
133 {
134 if (m_clientSocket == INVALID_SOCKET) return -1;
135
136 DmqHeader headerCopy = header;
137 auto payload = os.str();
138 if (payload.length() > UINT16_MAX) return -1;
139 headerCopy.SetLength(static_cast<uint16_t>(payload.length()));
140
141 dmq::xostringstream ss(std::ios::in | std::ios::out | std::ios::binary);
142
143 // Convert to Network Byte Order (Big Endian)
144 uint16_t marker = htons(headerCopy.GetMarker());
145 uint16_t id = htons(headerCopy.GetId());
146 uint16_t seq = htons(headerCopy.GetSeqNum());
147 uint16_t len = htons(headerCopy.GetLength());
148
149 ss.write((char*)&marker, 2);
150 ss.write((char*)&id, 2);
151 ss.write((char*)&seq, 2);
152 ss.write((char*)&len, 2);
153 ss.write(payload.data(), payload.size());
154
155 auto data = ss.str();
156 const char* ptr = data.c_str();
157 int remaining = (int)data.length();
158
159 while (remaining > 0)
160 {
161 int sent = send(m_clientSocket, ptr, remaining, 0);
162 if (sent == SOCKET_ERROR) return -1;
163 ptr += sent;
164 remaining -= sent;
165 }
166
167 // Always track the message (unless it is an ACK)
168 // Use Host Byte Order for ID check
169 if (headerCopy.GetId() != dmq::ACK_REMOTE_ID && m_transportMonitor)
170 m_transportMonitor->Add(headerCopy.GetSeqNum(), headerCopy.GetId());
171
172 return 0;
173 }

175 virtual int Receive(dmq::xstringstream& is, DmqHeader& header) override
176 {
177 // Lazy Accept Logic (Server)
178 if (m_type == Type::SERVER && m_clientSocket == INVALID_SOCKET)
179 {
180 TIMEVAL tv = { 0, 1000 }; // 1ms poll
181 fd_set fds;
182 FD_ZERO(&fds);
183 FD_SET(m_listenSocket, &fds);
184
185 if (select(0, &fds, NULL, NULL, &tv) > 0)
186 {
187 m_clientSocket = accept(m_listenSocket, NULL, NULL);
188 if (m_clientSocket != INVALID_SOCKET) {
189 std::cout << "Client accepted!" << std::endl;
190 DWORD timeout = 2000;
191 setsockopt(m_clientSocket, SOL_SOCKET, SO_RCVTIMEO, (const char*)&timeout, sizeof(timeout));
192 }
193 }
194 }
195
196 if (m_clientSocket == INVALID_SOCKET) return -1;
197
198 // Poll check to prevent blocking if no data is waiting (Critical for Sender polling ACKs)
199 fd_set readfds;
200 FD_ZERO(&readfds);
201 FD_SET(m_clientSocket, &readfds);
202 TIMEVAL tv = { 1, 0 }; // 1s poll to prevent busy loop
203
204 if (select(0, &readfds, NULL, NULL, &tv) <= 0) {
205 return -1;
206 }
207
208 // 1. Read Header
209 char headerBuf[DmqHeader::HEADER_SIZE];
210 if (!ReadExact(headerBuf, DmqHeader::HEADER_SIZE)) return -1;
211
212 dmq::xstringstream hs(std::ios::in | std::ios::out | std::ios::binary);
213 hs.write(headerBuf, DmqHeader::HEADER_SIZE);
214 hs.seekg(0);
215
216 uint16_t val;
217
218 // Read Marker (Convert Network -> Host)
219 hs.read((char*)&val, 2); header.SetMarker(ntohs(val));
220 if (header.GetMarker() != DmqHeader::MARKER) return -1;
221
222 hs.read((char*)&val, 2); header.SetId(ntohs(val));
223 hs.read((char*)&val, 2); header.SetSeqNum(ntohs(val));
224 hs.read((char*)&val, 2); header.SetLength(ntohs(val));
225
226 // 2. Read Payload
227 uint16_t length = header.GetLength();
228 if (length > 0)
229 {
230 std::vector<char> payload(length);
231 if (!ReadExact(payload.data(), length)) return -1;
232 is.write(payload.data(), length);
233 }
234
235 // 3. Ack
236 if (header.GetId() == dmq::ACK_REMOTE_ID) {
237 if (m_transportMonitor) m_transportMonitor->Remove(header.GetSeqNum());
238 }
239 else if (m_transportMonitor && m_sendTransport) {
240 dmq::xostringstream ss_ack;
241 DmqHeader ack;
242 ack.SetId(dmq::ACK_REMOTE_ID);
243 ack.SetSeqNum(header.GetSeqNum());
244 m_sendTransport->Send(ss_ack, ack);
245 }
246
247 return 0;
248 }
