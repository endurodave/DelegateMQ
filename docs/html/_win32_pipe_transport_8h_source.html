105 virtual int Send(dmq::xostringstream& os, const DmqHeader& header) override
106 {
107 if (os.bad() || os.fail())
108 return -1;
109
110 if (m_hPipe == INVALID_HANDLE_VALUE) return -1;
111
112 // Create a local copy to modify the length
113 DmqHeader headerCopy = header;
114
115 // Calculate payload size and set it on the copy
116 auto payload = os.str();
117 if (payload.length() > UINT16_MAX) {
118 std::cerr << "Error: Payload too large for 16-bit length." << std::endl;
119 return -1;
120 }
121 headerCopy.SetLength(static_cast<uint16_t>(payload.length()));
122
123 dmq::xostringstream ss(std::ios::in | std::ios::out | std::ios::binary);
124
125 // Use Network Byte Order (htons) for consistency
126 uint16_t marker = htons(headerCopy.GetMarker());
127 uint16_t id = htons(headerCopy.GetId());
128 uint16_t seqNum = htons(headerCopy.GetSeqNum());
129 uint16_t len = htons(headerCopy.GetLength());
130
131 ss.write(reinterpret_cast<const char*>(&marker), sizeof(marker));
132 ss.write(reinterpret_cast<const char*>(&id), sizeof(id));
133 ss.write(reinterpret_cast<const char*>(&seqNum), sizeof(seqNum));
134 ss.write(reinterpret_cast<const char*>(&len), sizeof(len));
135
136 // Insert delegate arguments (payload)
137 ss.write(payload.data(), payload.size());
138
139 auto fullPacket = ss.str();
140
141 DWORD sentLen = 0;
142 BOOL success = WriteFile(m_hPipe, fullPacket.c_str(), (DWORD)fullPacket.length(), &sentLen, NULL);
143
144 if (!success || sentLen != fullPacket.length())
145 return -1;
146
147 return 0;
148 }

150 virtual int Receive(dmq::xstringstream& is, DmqHeader& header) override
151 {
152 if (m_hPipe == INVALID_HANDLE_VALUE) return -1;
153
154 // Check/Accept connection
155 // Note: In non-blocking mode (PIPE_NOWAIT), this returns false immediately
156 // if connected, or error if already connected.
157 BOOL connected = ConnectNamedPipe(m_hPipe, NULL) ?
158 TRUE : (GetLastError() == ERROR_PIPE_CONNECTED);
159
160 // If not connected yet, we can't read.
161 if (connected == FALSE && GetLastError() == ERROR_NO_DATA) {
162 return -1;
163 }
164
165 DWORD size = 0;
166 BOOL success = ReadFile(m_hPipe, m_buffer, BUFFER_SIZE, &size, NULL);
167
168 if (success == FALSE || size <= 0)
169 return -1;
170
171 if (size <= DmqHeader::HEADER_SIZE) {
172 // std::cerr << "Received data is too small to process." << std::endl;
173 return -1;
174 }
175
176 dmq::xstringstream headerStream(std::ios::in | std::ios::out | std::ios::binary);
177 headerStream.write(m_buffer, DmqHeader::HEADER_SIZE);
178 headerStream.seekg(0);
179
180 uint16_t val = 0;
181
182 // Use Network Byte Order (ntohs)
183 headerStream.read(reinterpret_cast<char*>(&val), sizeof(val));
184 header.SetMarker(ntohs(val));
185
186 if (header.GetMarker() != DmqHeader::MARKER) {
187 return -1;
188 }
189
190 headerStream.read(reinterpret_cast<char*>(&val), sizeof(val));
191 header.SetId(ntohs(val));
192
193 headerStream.read(reinterpret_cast<char*>(&val), sizeof(val));
194 header.SetSeqNum(ntohs(val));
195
196 headerStream.read(reinterpret_cast<char*>(&val), sizeof(val));
197 header.SetLength(ntohs(val));
198
199 // Write the remaining argument data to stream
200 // Note: This relies on PIPE_TYPE_MESSAGE mode preserving boundaries.
201 // If 'size' is > HEADER_SIZE + payload length, we strictly write the payload part.
202 is.write(m_buffer + DmqHeader::HEADER_SIZE, size - DmqHeader::HEADER_SIZE);
203
204 return 0;
205 }
