135 virtual int Receive(dmq::xstringstream& is, DmqHeader& header) override
136 {
137 if (!m_port) return -1;
138 if (m_recvTransport != this) return -1;
139
140 char headerBuf[DmqHeader::HEADER_SIZE];
141
142 // 1. Strict Sync Loop
143 char b = 0;
144 while (true) {
145 // Short timeout for Sync, but non-blocking check
146 if (sp_blocking_read(m_port, &b, 1, 10) <= 0) return -1;
147
148 if ((uint8_t)b == 0xAA) {
149 char next_b = 0;
150 if (sp_blocking_read(m_port, &next_b, 1, 100) > 0) {
151 if ((uint8_t)next_b == 0x55) {
152 headerBuf[0] = b;
153 headerBuf[1] = next_b;
154 break;
155 }
156 }
157 }
158 }
159
160 // 2. Read Rest of Header
161 // INCREASED TIMEOUT: 1000ms (was 100ms) to allow OS latency
162 if (!ReadExact(headerBuf + 2, DmqHeader::HEADER_SIZE - 2, 1000))
163 return -1;
164
165 // 3. Deserialize
166 uint16_t val;
167 memcpy(&val, &headerBuf[0], 2); header.SetMarker(swap16(val));
168 memcpy(&val, &headerBuf[2], 2); header.SetId(swap16(val));
169 memcpy(&val, &headerBuf[4], 2); header.SetSeqNum(swap16(val));
170 memcpy(&val, &headerBuf[6], 2); header.SetLength(swap16(val));
171
172 if (header.GetMarker() != DmqHeader::MARKER) return -1;
173
174 // 4. Payload
175 uint16_t len = header.GetLength();
176 if (len > 0) {
177 if (len > BUFFER_SIZE) return -1;
178 // INCREASED TIMEOUT: 1000ms
179 if (!ReadExact(m_buffer, len, 1000)) return -1;
180 is.write(m_buffer, len);
181 }
182
183 // 5. CRC Check
184 uint16_t receivedCrc = 0;
185 // INCREASED TIMEOUT: 500ms
186 if (!ReadExact(reinterpret_cast<char*>(&receivedCrc), sizeof(receivedCrc), 500)) return -1;
187
188 uint16_t calcCrc = dmq::util::Crc16CalcBlock((unsigned char*)headerBuf, DmqHeader::HEADER_SIZE, 0xFFFF);
189 if (len > 0) calcCrc = dmq::util::Crc16CalcBlock((unsigned char*)m_buffer, len, calcCrc);
190
191 if (receivedCrc != calcCrc) {
192 std::cerr << "CRC Mismatch" << std::endl;
193 return -1;
194 }
195
196 // 6. ACKs
197 if (header.GetId() == dmq::ACK_REMOTE_ID) {
198 if (m_transportMonitor) m_transportMonitor->Remove(header.GetSeqNum());
199 }
200 else if (m_sendTransport) {
201 dmq::xostringstream ss_ack;
202 DmqHeader ack;
203 ack.SetId(dmq::ACK_REMOTE_ID);
204 ack.SetSeqNum(header.GetSeqNum());
205 m_sendTransport->Send(ss_ack, ack);
206 }
207 return 0;
208 }
