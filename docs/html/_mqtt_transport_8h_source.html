139 virtual int Send(xostringstream& os, const DmqHeader& header) override
140 {
141 if (os.bad() || os.fail())
142 return -1;
143
144 // Create a local copy to modify the length
145 DmqHeader headerCopy = header;
146 auto payload = os.str();
147 if (payload.length() > UINT16_MAX) {
148 std::cerr << "Payload too large." << std::endl;
149 return -1;
150 }
151 headerCopy.SetLength(static_cast<uint16_t>(payload.length()));
152
153 xostringstream ss(std::ios::in | std::ios::out | std::ios::binary);
154
155 // Serialize Header (Network Byte Order)
156 uint16_t marker = htons(headerCopy.GetMarker());
157 uint16_t id = htons(headerCopy.GetId());
158 uint16_t seqNum = htons(headerCopy.GetSeqNum());
159 uint16_t len = htons(headerCopy.GetLength());
160
161 ss.write(reinterpret_cast<const char*>(&marker), sizeof(marker));
162 ss.write(reinterpret_cast<const char*>(&id), sizeof(id));
163 ss.write(reinterpret_cast<const char*>(&seqNum), sizeof(seqNum));
164 ss.write(reinterpret_cast<const char*>(&len), sizeof(len));
165
166 // Append Payload
167 ss.write(payload.data(), payload.size());
168
169 auto fullPacket = ss.str();
170
171 MQTTClient_message pubmsg = MQTTClient_message_initializer;
172 MQTTClient_deliveryToken token;
173
174 pubmsg.payload = (void*)fullPacket.data();
175 pubmsg.payloadlen = (int)fullPacket.size();
176 pubmsg.qos = QOS;
177 pubmsg.retained = 0;
178
179 int rc;
180 if ((rc = MQTTClient_publishMessage(m_client, TOPIC, &pubmsg, &token)) != MQTTCLIENT_SUCCESS)
181 {
182 // printf("Failed to publish message, return code %d\n", rc);
183 return EXIT_FAILURE;
184 }
185 else
186 {
187 if (headerCopy.GetId() != dmq::ACK_REMOTE_ID && m_transportMonitor)
188 {
189 m_transportMonitor->Add(headerCopy.GetSeqNum(), headerCopy.GetId());
190 }
191 return EXIT_SUCCESS;
192 }
193 }

195 virtual int Receive(xstringstream& is, DmqHeader& header) override
196 {
197 std::unique_lock<std::mutex> lock(m_queueMtx);
198
199 // Wait for data or stop signal
200 if (m_queueCv.wait_for(lock, m_recvTimeout, [this] { return !m_rxQueue.empty() || m_stop; }))
201 {
202 if (m_stop) return -1;
203
204 if (m_rxQueue.empty()) return -1;
205
206 // Pop message
207 std::vector<char> msg = std::move(m_rxQueue.front());
208 m_rxQueue.pop();
209 lock.unlock(); // Release lock before processing
210
211 if (msg.size() < DmqHeader::HEADER_SIZE) return -1;
212
213 // Deserialize
214 xstringstream headerStream(std::ios::in | std::ios::out | std::ios::binary);
215 headerStream.write(msg.data(), DmqHeader::HEADER_SIZE);
216 headerStream.seekg(0);
217
218 uint16_t val;
219 headerStream.read((char*)&val, 2); header.SetMarker(ntohs(val));
220 if (header.GetMarker() != DmqHeader::MARKER) return -1;
221
222 headerStream.read((char*)&val, 2); header.SetId(ntohs(val));
223 headerStream.read((char*)&val, 2); header.SetSeqNum(ntohs(val));
224 headerStream.read((char*)&val, 2); header.SetLength(ntohs(val));
225
226 // Write payload
227 if (msg.size() > DmqHeader::HEADER_SIZE) {
228 is.write(msg.data() + DmqHeader::HEADER_SIZE, msg.size() - DmqHeader::HEADER_SIZE);
229 }
230
231 // ACK Logic
232 if (header.GetId() == dmq::ACK_REMOTE_ID) {
233 if (m_transportMonitor) m_transportMonitor->Remove(header.GetSeqNum());
234 }
235 else if (m_transportMonitor) {
236 xostringstream ss_ack;
237 DmqHeader ack;
238 ack.SetId(dmq::ACK_REMOTE_ID);
239 ack.SetSeqNum(header.GetSeqNum());
240 Send(ss_ack, ack);
241 }
242 return 0;
243 }
244
245 return -1; // Timeout
246 }

265 static int msgarrvd(void* context, char* topicName, int topicLen, MQTTClient_message* message)
266 {
267 MqttTransport* instance = (MqttTransport*)context;
268 if (instance)
269 {
270 // Copy data to vector and push to queue

272 data.resize(message->payloadlen);
273 std::memcpy(data.data(), message->payload, message->payloadlen);
274
275 {
276 std::lock_guard<std::mutex> lock(instance->m_queueMtx);
277 instance->m_rxQueue.push(std::move(data));
278 }
279 instance->m_queueCv.notify_one();

282 MQTTClient_freeMessage(&message);
