130 virtual int Send(xostringstream& os, const DmqHeader& header) override
131 {
132 if (os.bad() || os.fail())
133 return -1;
134
135 xostringstream ss(std::ios::in | std::ios::out | std::ios::binary);
136
137 // Write each header value using the getters from DmqHeader
138 auto marker = header.GetMarker();
139 ss << "[" << marker << " ";
140
141 auto id = header.GetId();
142 ss << id << " ";
143
144 auto seqNum = header.GetSeqNum();
145 ss << seqNum << "] ";
146
147 // Insert delegate arguments from the stream (os)
148 ss << os.rdbuf();
149
150 size_t payloadLen = ss.str().length();
151
152 // Allocate memory for the payload
153 void* payload = malloc(payloadLen); // Allocate memory to hold the payload
154 if (payload == NULL) {
155 std::cerr << "Failed to allocate memory for payload." << std::endl;
156 return -1; // Return error code if malloc fails
157 }
158
159 // Copy the content of the stringstream into the payload
160 std::memcpy(payload, ss.str().c_str(), payloadLen);
161
162 MQTTClient_message pubmsg = MQTTClient_message_initializer;
163 MQTTClient_deliveryToken token;
164 int rc;
165 pubmsg.payload = payload;
166 pubmsg.payloadlen = (int)payloadLen;
167 pubmsg.qos = MQTT_QOS;

188 virtual xstringstream Receive(DmqHeader& header) override
189 {
190 xstringstream headerStream(std::ios::in | std::ios::out | std::ios::binary);
191
192 if (!m_payload)
193 return headerStream;
194
195 // Find the position of the opening square bracket
196 size_t headerStart = 0;
197 while (m_payload[headerStart] != '[' && headerStart < m_payloadLen) {
198 ++headerStart;
199 }
200
201 if (headerStart == m_payloadLen) {
202 std::cerr << "Invalid header format: Missing opening '['" << std::endl;
203 return headerStream;
204 }
205
206 // Find the position of the closing square bracket
207 size_t headerEnd = headerStart;
208 while (m_payload[headerEnd] != ']' && headerEnd < m_payloadLen) {
209 ++headerEnd;
210 }
211
212 if (headerEnd == m_payloadLen) {
213 std::cerr << "Invalid header format: Missing closing ']'" << std::endl;
214 return headerStream;
215 }
216
217 // Calculate the header size (everything between '[' and ']')
218 size_t headerSize = headerEnd - headerStart + 1;
219
220 // Now extract the header between '[' and ']'
221 std::string headerStr(m_payload + headerStart + 1, headerSize - 2);
222
223 std::stringstream headerParser(headerStr);
224
225 uint16_t marker = 0, id = 0, seqNum = 0;
226 headerParser >> marker >> id >> seqNum;
227
228 header.SetMarker(marker);
229 header.SetId(id);
230 header.SetSeqNum(seqNum);
231
232 // Check for the correct sync marker
233 if (header.GetMarker() != DmqHeader::MARKER) {
234 std::cerr << "Invalid sync marker!" << std::endl;
235 return headerStream;
236 }
237
238 xstringstream argStream(std::ios::in | std::ios::out | std::ios::binary);
239
240 // The remaining data after the closing bracket is the JSON part
241 size_t jsonStart = headerEnd + 1; // After the closing header bracket ']'
242 argStream.write(m_payload + jsonStart, m_payloadLen - jsonStart);
243
244 if (id == dmq::ACK_REMOTE_ID)
245 {
246 // Receiver ack'ed message. Remove sequence number from monitor.
247 if (m_transportMonitor)
248 m_transportMonitor->Remove(seqNum);
249 }
250 else
251 {
252 if (m_transportMonitor)
253 {
254 // Send header with received seqNum as the ack message
255 xostringstream ss_ack;
256 DmqHeader ack;
257 ack.SetId(dmq::ACK_REMOTE_ID);
258 ack.SetSeqNum(seqNum);
259 Send(ss_ack, ack);
260 }
261 }
262
263 // argStream contains the serialized remote argument data
264 return argStream;
265 }
