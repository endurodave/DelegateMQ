132 bool RemoteInvokeWait(RemoteEndpoint<TClass, RetType(Args...)>& endpoint, Args&&... args)
133 {
134 // 1. [Caller Thread] Check if we are on the Network Thread.
135 if (Thread::GetCurrentThreadId() != m_thread.GetThreadId())
136 {
137 // 2. [Caller Thread] Create shared synchronization state.
138 struct SyncState {
139 std::atomic<bool> success{ false };
140 bool complete = false;
141 dmq::Mutex mtx; // Generic Mutex
142 dmq::ConditionVariable cv; // Generic CV
143 XALLOCATOR
144 };
145 auto state = std::make_shared<SyncState>();
146 dmq::DelegateRemoteId remoteId = endpoint.GetRemoteId();
147
148 // 3. [Caller Thread] Define the callback that wakes us up later.
149 std::function<void(dmq::DelegateRemoteId, uint16_t, TransportMonitor::Status)> statusCbFunc =
150 [state, remoteId](dmq::DelegateRemoteId id, uint16_t seq, TransportMonitor::Status status) {
151 if (id == remoteId) {
152 {
153 std::lock_guard<dmq::Mutex> lock(state->mtx);
154 state->complete = true;
155 if (status == TransportMonitor::Status::SUCCESS)
156 state->success.store(true);
157 }
158 // 9. [Network Thread] Notify the waiting caller thread.
159 state->cv.notify_one();
160 }
161 };
162
163 // 4. [Caller Thread] Register the callback.
164 dmq::ScopedConnection conn = m_transportMonitor.OnSendStatus->Connect(dmq::MakeDelegate(statusCbFunc));
165
166 // 5. [Caller Thread] Define the "Send" logic lambda.
167 auto* epPtr = &endpoint;
168 std::function<bool(Args...)> asyncCallFunc = [epPtr](Args... fwdArgs) -> bool {
169 // 7. [Network Thread] Execute the send operation.
170 (*epPtr)(fwdArgs...);
171 return (epPtr->GetError() == dmq::DelegateError::SUCCESS);
172 };
173
174 // 6. [Caller Thread] Dispatch the lambda to the Network Thread queue.
175 auto retVal = dmq::MakeDelegate(asyncCallFunc, m_thread, SEND_TIMEOUT)
176 .AsyncInvoke(std::forward<Args>(args)...);
177
178 if (retVal.has_value() && retVal.value() == true)
179 {
180 // 8. [Caller Thread] BLOCK and Wait.
181 std::unique_lock<dmq::Mutex> lock(state->mtx);
182 // wait_for returns false if the predicate is still false after the timeout
183 state->cv.wait_for(lock, RECV_TIMEOUT, [&] {
184 return state->complete;
185 });
186 // 10. [Caller Thread] Wake up! The wait is over.
187 }
188
189 // 11. [Caller Thread] Cleanup and return result.
190 // 'conn' goes out of scope here and automatically disconnects
191 return state->success.load();
192 }
193 else
194 {
195 // 12. [Network Thread] Alternative Path:
196 // We are already on the correct thread, so execute immediately.
197 endpoint(std::forward<Args>(args)...);
198 return (endpoint.GetError() == dmq::DelegateError::SUCCESS);
199 }
