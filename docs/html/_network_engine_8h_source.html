132 template <class TClass, class RetType, class... Args>
133 bool RemoteInvokeWait(dmq::DelegateMemberRemote<TClass, RetType(Args...)>& endpoint, Args&&... args)
134 {
135 // 1. [Caller Thread] Check if we are on the Network Thread.
136 if (Thread::GetCurrentThreadId() != m_thread.GetThreadId())
137 {
138 // 2. [Caller Thread] Create shared synchronization state.
139 struct SyncState {
140 std::atomic<bool> success{ false };
141 bool complete = false;
142 dmq::Mutex mtx; // Generic Mutex
143 dmq::ConditionVariable cv; // Generic CV
144 XALLOCATOR
145 };
146 auto state = xmake_shared<SyncState>();
147 dmq::DelegateRemoteId remoteId = endpoint.GetRemoteId();
148
149 // 3. [Caller Thread] Define the callback that wakes us up later.
150 std::function<void(dmq::DelegateRemoteId, uint16_t, TransportMonitor::Status)> statusCbFunc =
151 [state, remoteId](dmq::DelegateRemoteId id, uint16_t seq, TransportMonitor::Status status) {
152 if (id == remoteId) {
153 {
154 std::lock_guard<dmq::Mutex> lock(state->mtx);
155 state->complete = true;
156 if (status == TransportMonitor::Status::SUCCESS)
157 state->success.store(true);
158 }
159 // 9. [Network Thread] Notify the waiting caller thread.
160 state->cv.notify_one();
161 }
162 };
163
164 // 4. [Caller Thread] Register the callback.
165 dmq::ScopedConnection conn = m_transportMonitor.OnSendStatus.Connect(dmq::MakeDelegate(statusCbFunc));
166
167 // 5. [Caller Thread] Define the "Send" logic lambda.
168 auto* epPtr = &endpoint;
169 std::function<bool(Args...)> asyncCallFunc = [epPtr](auto&&... fwdArgs) -> bool {
170 // 7. [Network Thread] Execute the send operation.
171 (*epPtr)(std::forward<decltype(fwdArgs)>(fwdArgs)...);
172 return (epPtr->GetError() == dmq::DelegateError::SUCCESS);
173 };
174
175 // 6. [Caller Thread] Dispatch the lambda to the Network Thread queue.
176 auto retVal = dmq::MakeDelegate(asyncCallFunc, m_thread, SEND_TIMEOUT)
177 .AsyncInvoke(std::forward<Args>(args)...);
178
179 if (retVal.has_value() && retVal.value() == true)
180 {
181 // 8. [Caller Thread] BLOCK and Wait.
182 std::unique_lock<dmq::Mutex> lock(state->mtx);
183 // wait_for returns false if the predicate is still false after the timeout
184 state->cv.wait_for(lock, RECV_TIMEOUT, [&] {
185 return state->complete;
186 });
187 // 10. [Caller Thread] Wake up! The wait is over.
188 }
189
190 // 11. [Caller Thread] Cleanup and return result.
191 // 'conn' goes out of scope here and automatically disconnects
192 return state->success.load();
193 }
194 else
195 {
196 // 12. [Network Thread] Alternative Path:
197 // We are already on the correct thread, so execute immediately.
198 endpoint(std::forward<Args>(args)...);
199 return (endpoint.GetError() == dmq::DelegateError::SUCCESS);
200 }
201 }
