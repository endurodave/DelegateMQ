277 virtual RetType operator()(Args... args) override {
278 // Synchronously invoke the target function?
279 if (m_sync) {
280 if (this->Empty())
281 return RetType();
282
283 // Invoke the target function directly
284 return BaseType::operator()(std::forward<Args>(args)...);
285 }
286 else {
287 if (m_serializer) {
288 xostringstream arg_stream(std::ios::out | std::ios::in | std::ios::binary);
289
290 // Serialize all target function arguments into a stream
291 m_serializer->write(arg_stream, std::forward<Args>(args)...);
292
293 if (arg_stream.fail() || arg_stream.bad()) {
294 RaiseError(DelegateError::ERR_STREAM_FAIL);
295 }
296 else if (arg_stream.eof()) {
297 RaiseError(DelegateError::ERR_STREAM_EMPTY);
298 }
299 else {
300 // Dispatch delegate invocation to the remote destination
301 if (m_dispatcher) {
302 int error = m_dispatcher->Dispatch(arg_stream);
303 if (error)
304 RaiseError(DelegateError::ERR_DISPATCH_ERROR, error);
305 }
306 }
307 }

309 // Do not wait for remote to invoke function call
310 return RetType();

1121 // Synchronously invoke the target function?
1122 if (m_sync) {
1123 if (this->Empty())
1124 return RetType();
1125
1126 // Invoke the target function directly
1127 return BaseType::operator()(std::forward<Args>(args)...);
1128 }
1129 else {
1130 if (m_serializer) {
1131 xostringstream arg_stream(std::ios::out | std::ios::in | std::ios::binary);
1132
1133 // Serialize all target function arguments into a stream
1134 m_serializer->write(arg_stream, std::forward<Args>(args)...);
1135
1136 if (arg_stream.fail() || arg_stream.bad()) {
1137 RaiseError(DelegateError::ERR_STREAM_FAIL);
1138 }
1139 else if (arg_stream.eof()) {
1140 RaiseError(DelegateError::ERR_STREAM_EMPTY);
1141 }
1142 else {
1143 // Dispatch delegate invocation to the remote destination
1144 if (m_dispatcher) {
1145 int error = m_dispatcher->Dispatch(arg_stream);
1146 if (error)
1147 RaiseError(DelegateError::ERR_DISPATCH_ERROR, error);
1148 }
1149 }
1150 }
1151
1152 // Do not wait for remote to invoke function call
1153 return RetType();

1155 // Check if any argument is a shared_ptr with wrong usage
1156 // std::shared_ptr reference arguments are not allowed with asynchronous delegates as the behavior is
1157 // undefined. In other words:
1158 // void MyFunc(std::shared_ptr<T> data) // Ok!
1159 // void MyFunc(std::shared_ptr<T>& data) // Error
1160 static_assert(!(
1161 std::disjunction_v<is_shared_ptr_reference<Args>...>),
1162 "std::shared_ptr reference argument not allowed");
1163 }
