302 else {
303 // Dispatch delegate invocation to the remote destination
304 if (m_dispatcher) {
305 try {

1177 virtual RetType operator()(Args... args) override {
1178 // Synchronously invoke the target function?
1179 if (m_sync) {
1180 if (this->Empty())
1181 return RetType();
1182
1183 // Invoke the target function directly
1184 return BaseType::operator()(std::forward<Args>(args)...);
1185 }
1186 else {
1187 if (m_serializer && m_stream) {
1188 try {
1189 // Serialize all target function arguments into a stream
1190 m_serializer->Write(*m_stream, std::forward<Args>(args)...);
1191 } catch (std::exception&) {
1192 RaiseError(m_id, DelegateError::ERR_SERIALIZE);
1193 }
1194
1195 if (m_stream->fail() || m_stream->bad()) {
1196 RaiseError(m_id, DelegateError::ERR_STREAM_FAIL);
1197 }
1198 else if (m_stream->eof()) {
1199 RaiseError(m_id, DelegateError::ERR_STREAM_EMPTY);
1200 }
1201 else {
1202 // Dispatch delegate invocation to the remote destination
1203 if (m_dispatcher) {
1204 try {
1205 int error = m_dispatcher->Dispatch(*m_stream, m_id);
1206 if (error)
1207 RaiseError(m_id, DelegateError::ERR_DISPATCH, error);
1208 } catch (std::exception&) {
1209 RaiseError(m_id, DelegateError::ERR_DISPATCH);
1210 }
1211 } else {
1212 RaiseError(m_id, DelegateError::ERR_NO_DISPATCHER);
1213 }
1214 }
1215
1216 }
1217
1218 // Do not wait for remote to invoke function call
1219 return RetType();
1220
1221 // Check if any argument is a shared_ptr with wrong usage
1222 // std::shared_ptr reference arguments are not allowed with asynchronous delegates as the behavior is
1223 // undefined. In other words:
1224 // void MyFunc(std::shared_ptr<T> data) // Ok!
1225 // void MyFunc(std::shared_ptr<T>& data) // Error
1226 static_assert(!(
1227 std::disjunction_v<trait::is_shared_ptr_reference<Args>...>),
1228 "std::shared_ptr reference argument not allowed");
1229 }
1230 }
