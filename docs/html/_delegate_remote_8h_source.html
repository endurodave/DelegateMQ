278 virtual RetType operator()(Args... args) override {
279 // Synchronously invoke the target function?
280 if (m_sync) {
281 if (this->Empty())
282 return RetType();
283
284 // Invoke the target function directly
285 return BaseType::operator()(std::forward<Args>(args)...);
286 }
287 else {
288 if (m_serializer && m_stream) {
289 try {
290 // Serialize all target function arguments into a stream
291 m_serializer->Write(*m_stream, std::forward<Args>(args)...);
292 } catch (std::exception&) {
293 RaiseError(m_id, DelegateError::ERR_SERIALIZE);
294 }
295
296 if (!m_stream->good()) {
297 RaiseError(m_id, DelegateError::ERR_STREAM_NOT_GOOD);
298 }
299 else {
300 // Dispatch delegate invocation to the remote destination
301 if (m_dispatcher) {
302 try {
303 int error = m_dispatcher->Dispatch(*m_stream, m_id);
304 if (error)
305 RaiseError(m_id, DelegateError::ERR_DISPATCH, error);
306 } catch (std::exception&) {
307 RaiseError(m_id, DelegateError::ERR_DISPATCH);
308 }
309 } else {
310 RaiseError(m_id, DelegateError::ERR_NO_DISPATCHER);
311 }
312 }
