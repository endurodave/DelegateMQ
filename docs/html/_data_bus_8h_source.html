121 template <typename T, typename F>
122 dmq::ScopedConnection InternalSubscribe(const std::string& topic, F&& func, dmq::IThread* thread, QoS qos) {
123 SignalPtr<T> signal;
124
125 // Performance Note: Forced std::function construction for internal type management.
126 std::function<void(T)> typedFunc = std::forward<F>(func);
127 T* cachedValPtr = nullptr;
128 T cachedVal;
129 dmq::ScopedConnection conn;
130
131 {
132 std::lock_guard<dmq::RecursiveMutex> lock(m_mutex);
133
134 // 1. Enable LVC if requested (persists for topic lifetime until ResetForTesting)
135 if (qos.lastValueCache) {
136 m_topicQos[topic].lastValueCache = true;
137 }
138
139 // 2. Get or create signal with type safety check (std::type_index)
140 signal = GetOrCreateSignal<T>(topic);
141 if (!signal) {
142 return {}; // Type mismatch or other failure
143 }
144
145 // 3. Establish connection while holding the lock. This ensures no publishes
146 // are missed, as any new publish will be blocked until this lock is released.
147 if (thread) {
148 conn = signal->Connect(dmq::MakeDelegate(typedFunc, *thread));
149 } else {
150 conn = signal->Connect(dmq::MakeDelegate(typedFunc));
151 }
152
153 // 4. Prepare LVC delivery if enabled and available
154 if (qos.lastValueCache) {
155 auto it = m_lastValues.find(topic);
156 if (it != m_lastValues.end()) {
157 cachedVal = *std::static_pointer_cast<T>(it->second);
158 cachedValPtr = &cachedVal;
159 }
160 }
161 }
162
163 // 5. Dispatch LVC outside the lock to prevent deadlocks.
164 // IMPORTANT: Because this happens after releasing the lock, a high-frequency
165 // publisher on another thread could have already sent a new value to the
166 // connected signal. The subscriber might receive the fresh value FIRST,
167 // followed by this stale LVC value. Users of LVC should ensure their
168 // logic handles potential out-of-order state arrival.
169 if (cachedValPtr) {
170 if (thread) {
171 dmq::MakeDelegate(typedFunc, *thread).AsyncInvoke(*cachedValPtr);
172 } else {
173 typedFunc(*cachedValPtr);
174 }
175 }
176
177 return conn;
178 }

180 template <typename T>
181 void InternalPublish(const std::string& topic, const T& data) {
182 // Capture timestamp before lock acquisition for maximum accuracy and
183 // monotonic ordering using dmq::Clock.
184 auto now = dmq::Clock::now();
185 uint64_t timestamp = std::chrono::duration_cast<std::chrono::microseconds>(now.time_since_epoch()).count();
186
187 SignalPtr<T> signal;
188 std::shared_ptr<void> serializerPtr;
189 dmq::ISerializer<void(T)>* serializer = nullptr;
190 std::vector<std::shared_ptr<Participant>> participantsSnapshot;
191 std::string strVal = "?";
192 bool hasMonitor = false;
193
194 {
195 std::lock_guard<dmq::RecursiveMutex> lock(m_mutex);
196
197 // 1. Type safety: verify T matches the registered type for this topic.
198 // Must be first — before any writes — so a mismatch never corrupts LVC.
199 auto itType = m_typeIndices.find(topic);
200 if (itType != m_typeIndices.end() && itType->second != std::type_index(typeid(T))) {
201 ::FaultHandler(__FILE__, (unsigned short)__LINE__);
202 return;
203 }
204
205 // 2. Update LVC ONLY if enabled for this topic to save memory.
206 // NOTE: QoS lastValueCache is currently "sticky" per topic. Once enabled
207 // by any subscriber, it remains active for that topic until ResetForTesting().
208 auto itQos = m_topicQos.find(topic);
209 if (itQos != m_topicQos.end() && itQos->second.lastValueCache) {
210 m_lastValues[topic] = std::make_shared<T>(data);
211 }
212
213 // 3. Prepare monitor data
214 if (!m_monitorSignal.Empty()) {
215 hasMonitor = true;
216 auto itStr = m_stringifiers.find(topic);
217 if (itStr != m_stringifiers.end()) {
218 auto func = static_cast<std::function<std::string(const T&)>*>(itStr->second.get());
219 strVal = (*func)(data);
220 }
221 }
222
223 // 4. Get signal and remote info. Only create Signal if there is local interest.
224 auto itSig = m_signals.find(topic);
225 if (itSig != m_signals.end()) {
226 signal = std::static_pointer_cast<dmq::Signal<void(T)>>(itSig->second);
227 }
228
229 auto itSer = m_serializers.find(topic);
230 if (itSer != m_serializers.end()) {
231 serializerPtr = itSer->second;
232 serializer = static_cast<dmq::ISerializer<void(T)>*>(serializerPtr.get());
233 }
234
235 // 5. Snapshot participants while locked to ensure atomicity between
236 // local and remote dispatch sets.
237 participantsSnapshot = m_participants;
238 }
239
240 // 6. Dispatch Monitor outside lock to allow re-entry/prevent deadlocks
241 if (hasMonitor) {
242 SpyPacket packet{ topic, strVal, timestamp };
243 m_monitorSignal(packet);
244 }
245
246 // 7. Local distribution
247 if (signal) {
248 (*signal)(data);
249 }
250
251 // 8. Remote distribution using the snapshot
252 if (serializer) {
253 for (auto& participant : participantsSnapshot) {
254 participant->Send<T>(topic, data, *serializer);
255 }
256 }
257 }

316 template <typename T>
317 SignalPtr<T> GetOrCreateSignal(const std::string& topic) {
318 // Assume lock is held by caller
319 auto itType = m_typeIndices.find(topic);
320 if (itType != m_typeIndices.end()) {
321 if (itType->second != std::type_index(typeid(T))) {
322 // Runtime Type Safety: Catch same topic string used with different types
323 ::FaultHandler(__FILE__, (unsigned short)__LINE__);
324 return nullptr;
325 }
326 } else {
327 m_typeIndices.emplace(topic, std::type_index(typeid(T)));
328 }
329
330 auto it = m_signals.find(topic);
331 if (it != m_signals.end()) {
332 return std::static_pointer_cast<dmq::Signal<void(T)>>(it->second);
333 }
334
335 auto signal = std::make_shared<dmq::Signal<void(T)>>();
336 m_signals[topic] = std::static_pointer_cast<void>(signal);
337 return signal;
338 }

340 dmq::RecursiveMutex m_mutex;
341 std::unordered_map<std::string, std::shared_ptr<void>> m_signals;
342 std::unordered_map<std::string, std::type_index> m_typeIndices;
343 std::vector<std::shared_ptr<Participant>> m_participants;
344 std::unordered_map<std::string, std::shared_ptr<void>> m_serializers;
345 std::unordered_map<std::string, std::shared_ptr<void>> m_lastValues;
346 std::unordered_map<std::string, QoS> m_topicQos;
347 std::unordered_map<std::string, std::shared_ptr<void>> m_stringifiers;
348 dmq::Signal<void(const SpyPacket&)> m_monitorSignal;
