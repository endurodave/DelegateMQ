320
321 // Synchronously invoke the target function?
322 if (m_sync) {
323 // Invoke the target function directly
324 return BaseType::operator()(std::forward<Args>(args)...);
325 } else {
326 // Create a clone instance of this delegate

328 if (!delegate)

341 // will be called by the destination thread.
342 thread->DispatchDelegate(msg);
343

345 if (msg->GetSema().Wait(m_timeout)) {

347 const std::lock_guard<Mutex> lock(msg->GetLock());
