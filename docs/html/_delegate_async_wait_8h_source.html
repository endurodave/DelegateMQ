308 virtual RetType operator()(Args... args) override {
309 m_retVal.reset();
310 m_success = false;
311 if (this->Empty())
312 return RetType();
313
314 // Synchronously invoke the target function?
315 if (m_sync) {
316 // Invoke the target function directly
317 return BaseType::operator()(std::forward<Args>(args)...);
318 } else {
319 // Create a clone instance of this delegate
320 auto delegate = std::shared_ptr<ClassType>(Clone());
321 if (!delegate)
322 BAD_ALLOC();
323
324 // Create a new message instance for sending to the destination thread.
325 auto msg = std::make_shared<DelegateAsyncWaitMsg<Args...>>(delegate, std::forward<Args>(args)...);
326 if (!msg)
327 BAD_ALLOC();
328 msg->SetInvokerWaiting(true);
329
330 auto thread = this->GetThread();
331 if (thread) {
332 // Dispatch message onto the callback destination thread. Invoke()
333 // will be called by the destination thread.
334 thread->DispatchDelegate(msg);
335
336 // Wait for destination thread to execute the delegate function and get return value
337 if ((m_success = msg->GetSema().Wait(m_timeout)))
338 m_retVal = delegate->m_retVal;
339 }
340
341 // Protect data shared between source and destination threads
342 const std::lock_guard<std::mutex> lock(msg->GetLock());
343
344 // Set flag that source is not waiting anymore
345 msg->SetInvokerWaiting(false);
346
347 // Does the target function have a return value?
348 if constexpr (std::is_void<RetType>::value == false) {
349 // Is the return value valid?
350 if (m_retVal.has_value()) {
351 // Return the destination thread target function return value
352 return GetRetVal();
353 } else {
354 // Return a default return value
355 return RetType();
356 }
357 }
