317
318 // Synchronously invoke the target function?
319 if (m_sync) {
320 // Invoke the target function directly
321 return BaseType::operator()(std::forward<Args>(args)...);
322 } else {
323 // Create a clone instance of this delegate
324 auto delegate = std::shared_ptr<ClassType>(Clone());
325 if (!delegate)
326 BAD_ALLOC();

328 // Create a new message instance for sending to the destination thread.
329 auto msg = std::make_shared<DelegateAsyncWaitMsg<Args...>>(delegate, m_priority, std::forward<Args>(args)...);
330 if (!msg)

333
334 auto thread = this->GetThread();

336 // Dispatch message onto the callback destination thread. Invoke()
337 // will be called by the destination thread.
338 thread->DispatchDelegate(msg);
339
340 // Wait for destination thread to execute the delegate function and get return value
341 if ((m_success = msg->GetSema().Wait(m_timeout)))
342 m_retVal = delegate->m_retVal;
343 }

346 const std::lock_guard<std::mutex> lock(msg->GetLock());
347
348 // Set flag that source is not waiting anymore
349 msg->SetInvokerWaiting(false);

353 // Is the return value valid?

356 return GetRetVal();
