319 virtual RetType operator()(Args... args) override {
320 m_retVal.reset();
321 m_success = false;
322 if (this->Empty())
323 return RetType();
324
325 // Synchronously invoke the target function?
326 if (m_sync) {
327 // Invoke the target function directly
328 return BaseType::operator()(std::forward<Args>(args)...);
329 } else {
330 // Create a clone instance of this delegate
331 auto delegate = xmake_shared<ClassType>(*this);
332 if (!delegate)
333 BAD_ALLOC();
334
335 // Create a new message instance for sending to the destination thread.
336 auto msg = xmake_shared<DelegateAsyncWaitMsg<Args...>>(delegate, m_priority, std::forward<Args>(args)...);
337 if (!msg)
338 BAD_ALLOC();
339 msg->SetInvokerWaiting(true);
340
341 auto thread = this->GetThread();
342 if (thread) {
343 // Dispatch message onto the callback destination thread. Invoke()
344 // will be called by the destination thread.
345 thread->DispatchDelegate(msg);
346
347 // Wait for destination thread to execute the delegate function and get return value
348 if (msg->GetSema().Wait(m_timeout)) {
349 // Wait succeeded. Now acquire lock to safely read the value.
350 const dmq::LockGuard<Mutex> lock(msg->GetLock());
351 m_success = true;
352 m_retVal = delegate->m_retVal;
353 }
354 }
355
356 // Protect data shared between source and destination threads
357 const dmq::LockGuard<Mutex> lock(msg->GetLock());
358
359 // Set flag that source is not waiting anymore
360 msg->SetInvokerWaiting(false);
361
362 // Does the target function have a return value?
363 if constexpr (std::is_void<RetType>::value == false) {
364 // Is the return value valid?
365 if (m_retVal.has_value()) {
366 // Return the destination thread target function return value
367 return GetRetVal();
368 } else {
369 // Return a default return value
370 return RetType();
371 }
372 }
373 }
