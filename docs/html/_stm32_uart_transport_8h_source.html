57template <size_t Size>
58class UartRingBuffer {
59public:
61 bool Put(uint8_t data) {
62 size_t next_head = (m_head + 1) % Size;
63 if (next_head == m_tail) return false; // Buffer Full
64 m_buffer[m_head] = data;
65 m_head = next_head;
66 return true;
67 }
68
70 bool Get(uint8_t& data) {
71 if (m_head == m_tail) return false; // Buffer Empty
72 data = m_buffer[m_tail];
73 m_tail = (m_tail + 1) % Size;
74 return true;
75 }
76
77 void Clear() { m_head = m_tail = 0; }
78
79private:
80 volatile uint8_t m_buffer[Size];
81 volatile size_t m_head = 0;
82 volatile size_t m_tail = 0;
83};

132 virtual int Send(xostringstream& os, const DmqHeader& header) override {
133 if (os.bad() || os.fail()) return -1;
134 if (!m_huart || !m_mutex) return -1;
135
136 // CRITICAL SECTION: Protect UART TX hardware
137 if (xSemaphoreTakeRecursive(m_mutex, portMAX_DELAY) != pdTRUE) return -1;
138
139 // 1. Prepare Packet Structure
140 xstring payload = os.str();
141 DmqHeader headerCopy = header;
142 headerCopy.SetLength((uint16_t)payload.length());
143
144 uint8_t packet[DmqHeader::HEADER_SIZE];
145 auto to_net = [](uint16_t v) { return (uint16_t)((v << 8) | (v >> 8)); };
146
147 // Serialize Header (Network Byte Order)
148 uint16_t val;
149 val = to_net(headerCopy.GetMarker()); memcpy(&packet[0], &val, 2);
150 val = to_net(headerCopy.GetId()); memcpy(&packet[2], &val, 2);
151 val = to_net(headerCopy.GetSeqNum()); memcpy(&packet[4], &val, 2);
152 val = to_net(headerCopy.GetLength()); memcpy(&packet[6], &val, 2);
153
154 // Register with Monitor for Reliability (ACK tracking)
155 if (header.GetId() != dmq::ACK_REMOTE_ID && m_transportMonitor)
156 m_transportMonitor->Add(headerCopy.GetSeqNum(), headerCopy.GetId());
157
158 // 2. Send Header
159 if (HAL_UART_Transmit(m_huart, packet, DmqHeader::HEADER_SIZE, 100) != HAL_OK) {
160 xSemaphoreGiveRecursive(m_mutex); return -1;
161 }
162
163 // 3. Send Payload
164 if (payload.length() > 0) {
165 if (HAL_UART_Transmit(m_huart, (uint8_t*)payload.data(), (uint16_t)payload.length(), 500) != HAL_OK) {
166 xSemaphoreGiveRecursive(m_mutex); return -1;
167 }
168 }
169
170 // 4. Calculate and Send CRC (Required for PC communication)
171 uint16_t crc = Crc16CalcBlock(packet, DmqHeader::HEADER_SIZE, 0xFFFF);
172 if (payload.length() > 0) crc = Crc16CalcBlock((uint8_t*)payload.data(), payload.length(), crc);
173
174 if (HAL_UART_Transmit(m_huart, (uint8_t*)&crc, 2, 100) != HAL_OK) {
175 xSemaphoreGiveRecursive(m_mutex); return -1;
176 }
177
178 xSemaphoreGiveRecursive(m_mutex);
179 return 0;
180 }

182 // ------------------------------------------------------------------------
183 // Receive() Implementation
184 // ------------------------------------------------------------------------
185 virtual int Receive(xstringstream& is, DmqHeader& header) override {
186 if (m_recvTransport != this) return -1;
187
188 uint8_t headerBuf[DmqHeader::HEADER_SIZE];
189 uint8_t b = 0;
190 uint8_t markerHigh = (uint8_t)(DmqHeader::MARKER >> 8);
191
192 // 1. Sync Loop
193 // Reads from Ring Buffer. If empty, sleeps efficiently on Semaphore.
194 while (true) {
195 if (ReadByteBlocked(b)) {
196 if (b == markerHigh) { headerBuf[0] = b; break; }
197 }
198 }
199
200 // 2. Read Rest of Header
201 for (int i = 1; i < (int)DmqHeader::HEADER_SIZE; i++) {
202 if (!ReadByteBlocked(headerBuf[i])) return -1;
203 }
204
205 // Deserialize Header
206 auto from_net = [](uint16_t v) { return (uint16_t)((v << 8) | (v >> 8)); };
207 uint16_t val;
208 memcpy(&val, &headerBuf[0], 2); header.SetMarker(from_net(val));
209 memcpy(&val, &headerBuf[2], 2); header.SetId(from_net(val));
210 memcpy(&val, &headerBuf[4], 2); header.SetSeqNum(from_net(val));
211 memcpy(&val, &headerBuf[6], 2); header.SetLength(from_net(val));
212
213 if (header.GetMarker() != DmqHeader::MARKER) return -1;
214
215 // 3. Read Payload
216 uint16_t len = header.GetLength();
217 if (len > 0) {
218 if (len > BUFFER_SIZE) return -1;
219 for (int i = 0; i < len; i++) {
220 if (!ReadByteBlocked(m_tempRxBuffer[i])) return -1;
221 }
222 is.clear(); is.str(""); is.write((char*)m_tempRxBuffer, len);
223 }
224
225 // 4. Read & Consume CRC
226 uint16_t receivedCrc;
227 uint8_t* pCrc = (uint8_t*)&receivedCrc;
228 if (!ReadByteBlocked(pCrc[0])) return -1;
229 if (!ReadByteBlocked(pCrc[1])) return -1;
230
231 // 5. Handle ACKs
232 if (header.GetId() == dmq::ACK_REMOTE_ID) {
233 if (m_transportMonitor) m_transportMonitor->Remove(header.GetSeqNum());
234 }
235 else if (m_transportMonitor && m_sendTransport) {
236 // Auto-Reply with ACK
237 xostringstream ss;
238 DmqHeader ack;
239 ack.SetId(dmq::ACK_REMOTE_ID);
240 ack.SetSeqNum(header.GetSeqNum());
241 m_sendTransport->Send(ss, ack);
242 }
243 return 0;
244 }
