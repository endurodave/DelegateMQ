64 int Create(Type type, const char* addr, uint16_t port)
65 {
66 m_type = type;
67 m_socket = socket(AF_INET, SOCK_STREAM, 0);
68 if (m_socket < 0) return -1;
69
70 // Set TCP_NODELAY to disable Nagle's algorithm for low-latency RPC
71 int flag = 1;
72 setsockopt(m_socket, IPPROTO_TCP, TCP_NODELAY, (char*)&flag, sizeof(int));
73
74 sockaddr_in srv_addr{};
75 srv_addr.sin_family = AF_INET;
76 srv_addr.sin_port = htons(port);
77 inet_aton(addr, &srv_addr.sin_addr);
78
79 if (type == Type::SERVER) {
80 int opt = 1;
81 setsockopt(m_socket, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));
82 if (bind(m_socket, (struct sockaddr*)&srv_addr, sizeof(srv_addr)) < 0) return -1;
83 if (listen(m_socket, 5) < 0) return -1;
84 }
85 else {
86 if (connect(m_socket, (struct sockaddr*)&srv_addr, sizeof(srv_addr)) < 0) return -1;
87 m_connFd = m_socket;
88
89 struct timeval tv;
90 tv.tv_sec = 2; tv.tv_usec = 0;
91 setsockopt(m_connFd, SOL_SOCKET, SO_RCVTIMEO, (const char*)&tv, sizeof(tv));
92 }
93
94 return 0;
95 }

181 int SendToSocket(int fd, xostringstream& os, const DmqHeader& header) {
182 if (fd < 0) return -1;
183
184 auto payload = os.str();
185 DmqHeader headerCopy = header;
186 headerCopy.SetLength(static_cast<uint16_t>(payload.length()));
187
188 xostringstream ss(std::ios::binary);
189
190 // Convert to Network Byte Order (Big Endian)
191 uint16_t marker = htons(headerCopy.GetMarker());
192 uint16_t id = htons(headerCopy.GetId());
193 uint16_t seqNum = htons(headerCopy.GetSeqNum());
194 uint16_t length = htons(headerCopy.GetLength());
195
196 ss.write((char*)&marker, 2);
197 ss.write((char*)&id, 2);
198 ss.write((char*)&seqNum, 2);
199 ss.write((char*)&length, 2);
200 ss.write(payload.data(), payload.size());
201
202 auto packet = ss.str();
203
204 ssize_t sent = write(fd, packet.data(), packet.size());
205 if (sent != (ssize_t)packet.size()) return -1;
206
207 if (headerCopy.GetId() != dmq::ACK_REMOTE_ID && m_transportMonitor)
208 m_transportMonitor->Add(headerCopy.GetSeqNum(), headerCopy.GetId());
209
210 return 0;
211 }

214 int ReceiveServer(xstringstream& is, DmqHeader& header) {
215 // 1. Lazy Accept: Check for new client connections
216 fd_set fds;
217 FD_ZERO(&fds);
218 FD_SET(m_socket, &fds);
219 struct timeval tv = { 0, 1000 }; // 1ms poll
220
221 if (select(m_socket + 1, &fds, nullptr, nullptr, &tv) > 0) {
222 int client = accept(m_socket, nullptr, nullptr);
223 if (client >= 0) {
224 struct timeval t;
225 t.tv_sec = m_recvTimeout.count() / 1000;
226 t.tv_usec = (m_recvTimeout.count() % 1000) * 1000;
227 setsockopt(client, SOL_SOCKET, SO_RCVTIMEO, &t, sizeof(t));
228 m_serverClients.push_back(client);
229 }
230 }
231
232 // 2. Poll all connected clients for data
233 if (m_serverClients.empty()) return -1;
234
235 fd_set readfds;
236 FD_ZERO(&readfds);
237 int max_fd = -1;
238 for (auto s : m_serverClients) {
239 FD_SET(s, &readfds);
240 if (s > max_fd) max_fd = s;
241 }
242
243 tv.tv_usec = 1000; // 1ms poll
244 if (select(max_fd + 1, &readfds, nullptr, nullptr, &tv) > 0) {
245 for (auto it = m_serverClients.begin(); it != m_serverClients.end(); ) {
246 if (FD_ISSET(*it, &readfds)) {
247 int result = ReceiveSocket(*it, is, header);
248 if (result == -2) { // Disconnected
249 close(*it);
250 it = m_serverClients.erase(it);
251 continue;
252 }
253 return result;
254 }
255 ++it;
256 }
257 }
258 return -1;
259 }

262 int ReceiveSocket(int fd, xstringstream& is, DmqHeader& header) {
263 if (fd < 0) return -1;
264
265 fd_set readfds;
266 FD_ZERO(&readfds);
267 FD_SET(fd, &readfds);
268 struct timeval tv = { 0, 1000 };
269 if (select(fd + 1, &readfds, nullptr, nullptr, &tv) <= 0) return -1;
270
271 // 1. Read Header
272 char headerBuf[DmqHeader::HEADER_SIZE];
273 if (!ReadExact(fd, headerBuf, DmqHeader::HEADER_SIZE)) return -2; // Disconnected
274
275 xstringstream ss(std::ios::in | std::ios::out | std::ios::binary);
276 ss.write(headerBuf, DmqHeader::HEADER_SIZE);
277 ss.seekg(0);
278
279 uint16_t val;
280
281 // Read Marker (Convert Network -> Host)
282 ss.read((char*)&val, 2); header.SetMarker(ntohs(val));
283 if (header.GetMarker() != DmqHeader::MARKER) return -1;
284
285 ss.read((char*)&val, 2); header.SetId(ntohs(val));
286 ss.read((char*)&val, 2); header.SetSeqNum(ntohs(val));
287 ss.read((char*)&val, 2); header.SetLength(ntohs(val));
288
289 // 2. Read Payload
290 uint16_t length = header.GetLength();
291 if (length > 0) {
292 m_recvBuf.resize(length);
293 if (!ReadExact(fd, m_recvBuf.data(), length)) return -2;
294 is.write(m_recvBuf.data(), length);
295 }
296
297 // 3. Handle Acknowledgment
298 if (header.GetId() == dmq::ACK_REMOTE_ID) {
299 if (m_transportMonitor) m_transportMonitor->Remove(header.GetSeqNum());
300 }
301 else if (m_sendTransport) {
302 xostringstream ss_ack;
303 DmqHeader ack;
304 ack.SetId(dmq::ACK_REMOTE_ID);
305 ack.SetSeqNum(header.GetSeqNum());
306 m_sendTransport->Send(ss_ack, ack);
307 }
308 return 0;
309 }

312 bool ReadExact(int fd, char* buf, size_t len)
313 {
314 size_t total = 0;
315 while (total < len) {
316 ssize_t r = read(fd, buf + total, len - total);
317 if (r <= 0) return false;
318 total += r;
319 }
320 return true;
321 }
