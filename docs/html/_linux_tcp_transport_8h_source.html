55 if (m_socket < 0) return -1;
56
57 // Set TCP_NODELAY to disable Nagle's algorithm for low-latency RPC
58 int flag = 1;
59 setsockopt(m_socket, IPPROTO_TCP, TCP_NODELAY, (char*)&flag, sizeof(int));
60

62 srv_addr.sin_family = AF_INET;
63 srv_addr.sin_port = htons(port);

65

68 setsockopt(m_socket, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));
69 if (bind(m_socket, (struct sockaddr*)&srv_addr, sizeof(srv_addr)) < 0) return -1;
70 if (listen(m_socket, 1) < 0) return -1;
71
72 // We will accept lazily in Receive().
73 }

77 }

117 virtual int Send(xostringstream& os, const DmqHeader& header) override
118 {
119 if (m_connFd < 0) return -1;
120
121 std::string payload = os.str();
122 DmqHeader headerCopy = header;
123 headerCopy.SetLength(static_cast<uint16_t>(payload.length()));
124
125 xostringstream ss(std::ios::binary);
126
127 // Convert to Network Byte Order (Big Endian)
128 uint16_t marker = htons(headerCopy.GetMarker());
129 uint16_t id = htons(headerCopy.GetId());
130 uint16_t seqNum = htons(headerCopy.GetSeqNum());
131 uint16_t length = htons(headerCopy.GetLength());
132
133 ss.write((char*)&marker, 2);
134 ss.write((char*)&id, 2);
135 ss.write((char*)&seqNum, 2);
136 ss.write((char*)&length, 2);
137 ss.write(payload.data(), payload.size());
138
139 std::string packet = ss.str();
140
141 // Always track the message (unless it is an ACK)
142 // Use Host Byte Order for ID check
143 if (headerCopy.GetId() != dmq::ACK_REMOTE_ID && m_transportMonitor)
144 m_transportMonitor->Add(headerCopy.GetSeqNum(), headerCopy.GetId());
145
146 // write() is thread-safe on Linux sockets
147 ssize_t sent = write(m_connFd, packet.data(), packet.size());
148 return (sent == (ssize_t)packet.size()) ? 0 : -1;

151 virtual int Receive(xstringstream& is, DmqHeader& header) override
152 {
153 // Lazy Accept Logic (Server Mode)
154 if (m_type == Type::SERVER && m_connFd < 0) {
155 fd_set fds;
156 FD_ZERO(&fds);
157 FD_SET(m_socket, &fds);
158 struct timeval tv = { 0, 1000 }; // 1ms poll
159
160 if (select(m_socket + 1, &fds, nullptr, nullptr, &tv) > 0) {
161 m_connFd = accept(m_socket, nullptr, nullptr);
162 if (m_connFd >= 0) {
163 // Set timeout on new socket
164 struct timeval t; t.tv_sec = 2; t.tv_usec = 0;
165 setsockopt(m_connFd, SOL_SOCKET, SO_RCVTIMEO, (const char*)&t, sizeof(t));
166 }
167 }
168 }
169
170 if (m_connFd < 0) return -1;
171
172 // --- Poll check to prevent blocking if no data is waiting ---
173 fd_set readfds;
174 FD_ZERO(&readfds);
175 FD_SET(m_connFd, &readfds);
176 struct timeval tv;
177 tv.tv_sec = 1; // Increased to 1s to prevent busy loop
178 tv.tv_usec = 0;
179
180 // If no data is waiting, return -1 immediately so Sender loop continues
181 if (select(m_connFd + 1, &readfds, NULL, NULL, &tv) <= 0) {
182 return -1;
183 }
184
185 // 1. Read Fixed-Size Header for Framing
186 char headerBuf[DmqHeader::HEADER_SIZE];
187 if (!ReadExact(headerBuf, DmqHeader::HEADER_SIZE)) return -1;
188
189 xstringstream ss(std::ios::in | std::ios::out | std::ios::binary);
190 ss.write(headerBuf, DmqHeader::HEADER_SIZE);
191 ss.seekg(0);
192
193 uint16_t val;
194
195 // Read Marker (Convert Network -> Host)
196 ss.read((char*)&val, 2); header.SetMarker(ntohs(val));
197 if (header.GetMarker() != DmqHeader::MARKER) return -1;
198
199 ss.read((char*)&val, 2); header.SetId(ntohs(val));
200 ss.read((char*)&val, 2); header.SetSeqNum(ntohs(val));
201 ss.read((char*)&val, 2); header.SetLength(ntohs(val));
202
203 // 2. Read Payload based on Header Length
204 uint16_t length = header.GetLength();
205 if (length > 0) {
206 std::vector<char> payload(length);
207 if (!ReadExact(payload.data(), length)) return -1;
208 is.write(payload.data(), length);
209 }
210
211 // 3. Reliability Logic
212 if (header.GetId() == dmq::ACK_REMOTE_ID) {
213 if (m_transportMonitor) m_transportMonitor->Remove(header.GetSeqNum());
214 }
215 else if (m_sendTransport) {
216 xostringstream ss_ack;
217 DmqHeader ack;
218 ack.SetId(dmq::ACK_REMOTE_ID);
219 ack.SetSeqNum(header.GetSeqNum());
220 m_sendTransport->Send(ss_ack, ack);
221 }
222 return 0;
223 }
