105 // --- PORTABLE LOCK GUARD ---
106 // Does not require <mutex>. Works with any BasicLockable type (lock/unlock).
107 template<typename T>
108 class PortableLockGuard {
109 T& m_mutex;
110 public:
111 explicit PortableLockGuard(T& m) noexcept : m_mutex(m) { m_mutex.lock(); }
112 ~PortableLockGuard() noexcept { m_mutex.unlock(); }
113 PortableLockGuard(const PortableLockGuard&) = delete;
114 PortableLockGuard& operator=(const PortableLockGuard&) = delete;
115 };
