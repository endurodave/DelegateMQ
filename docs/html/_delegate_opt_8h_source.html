98 // --- PORTABLE LOCK GUARD ---
99 // Does not require <mutex>. Works with any BasicLockable type (lock/unlock).
100 template<typename T>
101 class PortableLockGuard {
102 T& m_mutex;
103 public:
104 explicit PortableLockGuard(T& m) noexcept : m_mutex(m) { m_mutex.lock(); }
105 ~PortableLockGuard() noexcept { m_mutex.unlock(); }
106 PortableLockGuard(const PortableLockGuard&) = delete;
107 PortableLockGuard& operator=(const PortableLockGuard&) = delete;
108 };
