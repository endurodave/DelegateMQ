189 virtual int Send(dmq::xostringstream& os, const DmqHeader& header) override
190 {
191 std::lock_guard<std::recursive_mutex> lock(m_mutex);
192
193 if (os.bad() || os.fail()) {
194 std::cout << "Error: xostringstream is in a bad state!" << std::endl;
195 return -1;
196 }
197
198 if (m_type == Type::SUB) {
199 std::cout << "Error: Cannot send on SUB socket!" << std::endl;
200 return -1;
201 }
202
203 if (m_sendTransport != this) {
204 std::cout << "Error: This transport used for receive only!" << std::endl;
205 return -1;
206 }
207
208 // Create a local copy to modify the length
209 DmqHeader headerCopy = header;
210
211 // Get payload and set length on the copy
212 auto payload = os.str();
213 if (payload.length() > UINT16_MAX) {
214 std::cerr << "Error: Payload too large for 16-bit length." << std::endl;
215 return -1;
216 }
217 headerCopy.SetLength(static_cast<uint16_t>(payload.length()));
218
219 dmq::xostringstream ss(std::ios::in | std::ios::out | std::ios::binary);
220
221 // Write header values from the COPY (using htons)
222 uint16_t marker = htons(headerCopy.GetMarker());
223 ss.write(reinterpret_cast<const char*>(&marker), sizeof(marker));
224
225 uint16_t id = htons(headerCopy.GetId());
226 ss.write(reinterpret_cast<const char*>(&id), sizeof(id));
227
228 uint16_t seqNum = htons(headerCopy.GetSeqNum());
229 ss.write(reinterpret_cast<const char*>(&seqNum), sizeof(seqNum));
230
231 uint16_t len = htons(headerCopy.GetLength());
232 ss.write(reinterpret_cast<const char*>(&len), sizeof(len));
233
234 // Insert delegate arguments (payload)
235 ss.write(payload.data(), payload.size());
236
237 auto fullPacket = ss.str();
238
239 // Send data using NNG
240 // Use NNG_FLAG_NONBLOCK to prevent deadlocks if peer is gone
241 int err = nng_send(m_nngSocket, (void*)fullPacket.data(), fullPacket.size(), NNG_FLAG_NONBLOCK);
242 if (err != 0)
243 {
244 // NNG_EAGAIN means queue full (congestion), treat as error to trigger retry logic
245 if (err != NNG_EAGAIN) {
246 // std::cerr << "nng_send failed with error: " << nng_strerror(err) << std::endl;
247 }
248 return -1;
249 }
250
251 if (headerCopy.GetId() != dmq::ACK_REMOTE_ID)
252 {
253 if (m_transportMonitor)
254 m_transportMonitor->Add(headerCopy.GetSeqNum(), headerCopy.GetId());
255 }
256
257 return 0;
258 }

260 virtual int Receive(dmq::xstringstream& is, DmqHeader& header) override
261 {
262 // Lock Guard
263 std::lock_guard<std::recursive_mutex> lock(m_mutex);
264
265 if (nng_socket_id(m_nngSocket) == 0) {
266 // std::cout << "Error: Socket not created!" << std::endl;
267 return -1;
268 }
269
270 if (m_type == Type::PUB) {
271 std::cout << "Error: Cannot receive on PUB socket!" << std::endl;
272 return -1;
273 }
274
275 if (m_recvTransport != this) {
276 std::cout << "Error: This transport used for send only!" << std::endl;
277 return -1;
278 }
279
280 dmq::xstringstream headerStream(std::ios::in | std::ios::out | std::ios::binary);
281
282 size_t size = sizeof(m_buffer);
283
284 // Receive with 100ms timeout (set in Create)
285 int err = nng_recv(m_nngSocket, m_buffer, &size, 0);
286 if (err != 0) {
287 if (err == NNG_ETIMEDOUT) {
288 // Just a timeout, not a fatal error
289 return -1;
290 }
291 // std::cerr << "nng_recv failed with error: " << nng_strerror(err) << std::endl;
292 return -1;
293 }
294
295 if (size < DmqHeader::HEADER_SIZE) {
296 std::cerr << "Received data is too small to process." << std::endl;
297 return -1;
298 }
299
300 // Write the received header data into the stringstream
301 headerStream.write(m_buffer, DmqHeader::HEADER_SIZE);
302 headerStream.seekg(0);
303
304 uint16_t val = 0;
305
306 // Read Marker (Convert Network -> Host)
307 headerStream.read(reinterpret_cast<char*>(&val), sizeof(val));
308 header.SetMarker(ntohs(val));
309
310 if (header.GetMarker() != DmqHeader::MARKER) {
311 std::cerr << "Invalid sync marker!" << std::endl;
312 return -1; // @TODO: Optionally handle this case more gracefully
313 }
314
315 // Read the DelegateRemoteId (2 bytes) into the `id` variable
316 headerStream.read(reinterpret_cast<char*>(&val), sizeof(val));
317 header.SetId(ntohs(val));
318
319 // Read seqNum using the getter for byte swapping
320 headerStream.read(reinterpret_cast<char*>(&val), sizeof(val));
321 header.SetSeqNum(ntohs(val));
322
323 // Read length using the getter for byte swapping
324 headerStream.read(reinterpret_cast<char*>(&val), sizeof(val));
325 header.SetLength(ntohs(val));
326
327 // Write the remaining target function argument data to stream
328 is.write(m_buffer + DmqHeader::HEADER_SIZE, size - DmqHeader::HEADER_SIZE);
329
330 if (header.GetId() == dmq::ACK_REMOTE_ID)
331 {
332 // Receiver ack'ed message. Remove sequence number from monitor.
333 if (m_transportMonitor)
334 m_transportMonitor->Remove(header.GetSeqNum());
335 }
336 else
337 {
338 if (m_transportMonitor && m_sendTransport)
339 {
340 // Send header with received seqNum as the ack message
341 dmq::xostringstream ss_ack;
342 DmqHeader ack;
343 ack.SetId(dmq::ACK_REMOTE_ID);
344 ack.SetSeqNum(header.GetSeqNum());
345
346 // Note: Recursive mutex allows Send() here
347 m_sendTransport->Send(ss_ack, ack);
348 }
349 }
350
351 return 0; // Success
352 }
