117 std::lock_guard<dmq::RecursiveMutex> lock(m_mutex);
118 auto it = m_channels.find(remoteId);

120 // Type safety: catch remoteId reused with a different T
121 auto itType = m_channelTypes.find(remoteId);
122 if (itType != m_channelTypes.end() && itType->second != std::type_index(typeid(T))) {
123 ::FaultHandler(__FILE__, (unsigned short)__LINE__);
124 return nullptr;
125 }

127 }
128
129 auto channel = std::make_shared<dmq::RemoteChannel<void(T)>>(*m_transport, serializer);
130
131 // Establish the remote ID for sending via operator().
132 channel->SetRemoteId(remoteId);
133
134 m_channels[remoteId] = { channel, channel->GetEndpoint() };
135 m_channelTypes.emplace(remoteId, std::type_index(typeid(T)));
