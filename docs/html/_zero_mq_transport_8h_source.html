194 virtual int Receive(xstringstream& is, DmqHeader& header) override
195 {
196 if (Thread::GetCurrentThreadId() != m_thread.GetThreadId())

201 return -1;

205 std::cout << "Error: Cannot receive on PUB socket!" << std::endl;

214 xstringstream headerStream(std::ios::in | std::ios::out | std::ios::binary);
215
216 int size = zmq_recv(m_zmq, m_buffer, BUFFER_SIZE, 0);
217 if (size == -1) {
218 // Check if the error is due to a timeout
219 if (zmq_errno() == EAGAIN) {
220 //std::cout << "Receive timeout!" << std::endl;
221 }
222 return -1;
223 }
224
225 if (size < DmqHeader::HEADER_SIZE) {
226 std::cerr << "Received data is too small to process." << std::endl;
227 return -1;
228 }
229
230 // Write the received header data into the stringstream
231 headerStream.write(m_buffer, DmqHeader::HEADER_SIZE);
232 headerStream.seekg(0);
233
234 uint16_t marker = 0;
235 headerStream.read(reinterpret_cast<char*>(&marker), sizeof(marker));
236 header.SetMarker(marker);
237
238 if (header.GetMarker() != DmqHeader::MARKER) {
239 std::cerr << "Invalid sync marker!" << std::endl;
240 return -1; // @TODO: Optionally handle this case more gracefully
241 }
242
243 // Read the DelegateRemoteId (2 bytes) into the `id` variable
244 uint16_t id = 0;
245 headerStream.read(reinterpret_cast<char*>(&id), sizeof(id));
246 header.SetId(id);
247
248 // Read seqNum using the getter for byte swapping
249 uint16_t seqNum = 0;
250 headerStream.read(reinterpret_cast<char*>(&seqNum), sizeof(seqNum));
251 header.SetSeqNum(seqNum);
252
253 // Write the remaining target function argument data to stream
254 is.write(m_buffer + DmqHeader::HEADER_SIZE, size - DmqHeader::HEADER_SIZE);
255
256 if (id == dmq::ACK_REMOTE_ID)
257 {
258 // Receiver ack'ed message. Remove sequence number from monitor.
259 if (m_transportMonitor)
260 m_transportMonitor->Remove(seqNum);
261 }
262 else
263 {
264 if (m_transportMonitor && m_sendTransport)
265 {
266 // Send header with received seqNum as the ack message
267 xostringstream ss_ack;
268 DmqHeader ack;
269 ack.SetId(dmq::ACK_REMOTE_ID);
270 ack.SetSeqNum(seqNum);
271 m_sendTransport->Send(ss_ack, ack);
272 }
273 }
274
