143

193 virtual int Receive(xstringstream& is, DmqHeader& header) override
194 {
195 if (Thread::GetCurrentThreadId() != m_thread.GetThreadId())
196 return MakeDelegate(this, &ZeroMqTransport::Receive, m_thread, dmq::WAIT_INFINITE)(is, header);
197
198 if (m_zmq == nullptr) {
199 std::cout << "Error: Socket not created!" << std::endl;
200 return -1;
201 }
202
203 if (m_type == Type::PUB) {
204 std::cout << "Error: Cannot receive on PUB socket!" << std::endl;
205 return -1;
206 }
207
208 if (m_recvTransport != this) {
209 std::cout << "Error: This transport used for send only!" << std::endl;
210 return -1;

213 xstringstream headerStream(std::ios::in | std::ios::out | std::ios::binary);
214
215 int size = zmq_recv(m_zmq, m_buffer, BUFFER_SIZE, 0);
216 if (size == -1) {
217 // Check if the error is due to a timeout
218 if (zmq_errno() == EAGAIN) {
219 //std::cout << "Receive timeout!" << std::endl;
220 }
221 return -1;
222 }
223
224 if (size < DmqHeader::HEADER_SIZE) {
225 std::cerr << "Received data is too small to process." << std::endl;
226 return -1;
227 }
228
229 // Write the received header data into the stringstream
230 headerStream.write(m_buffer, DmqHeader::HEADER_SIZE);
231 headerStream.seekg(0);
232
233 uint16_t marker = 0;
234 headerStream.read(reinterpret_cast<char*>(&marker), sizeof(marker));
235 header.SetMarker(marker);
236
237 if (header.GetMarker() != DmqHeader::MARKER) {
238 std::cerr << "Invalid sync marker!" << std::endl;
239 return -1; // TODO: Optionally handle this case more gracefully
240 }
241
242 // Read the DelegateRemoteId (2 bytes) into the `id` variable
243 uint16_t id = 0;
244 headerStream.read(reinterpret_cast<char*>(&id), sizeof(id));
245 header.SetId(id);
246
247 // Read seqNum using the getter for byte swapping
248 uint16_t seqNum = 0;
249 headerStream.read(reinterpret_cast<char*>(&seqNum), sizeof(seqNum));
250 header.SetSeqNum(seqNum);
251
252 // Write the remaining target function argument data to stream
253 is.write(m_buffer + DmqHeader::HEADER_SIZE, size - DmqHeader::HEADER_SIZE);
254
255 if (id == dmq::ACK_REMOTE_ID)
256 {
257 // Receiver ack'ed message. Remove sequence number from monitor.
258 if (m_transportMonitor)
259 m_transportMonitor->Remove(seqNum);
260 }
261 else
262 {
263 if (m_transportMonitor && m_sendTransport)
264 {
265 // Send header with received seqNum as the ack message
266 xostringstream ss_ack;
267 DmqHeader ack;
268 ack.SetId(dmq::ACK_REMOTE_ID);
269 ack.SetSeqNum(seqNum);
270 m_sendTransport->Send(ss_ack, ack);
271 }
272 }
273
274 return 0; // Success
275 }
