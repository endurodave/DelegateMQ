171 virtual int Send(xostringstream& os, const DmqHeader& header) override
172 {
173 std::lock_guard<std::recursive_mutex> lock(m_mutex);
174
175 if (os.bad() || os.fail()) {
176 std::cout << "Error: xostringstream is in a bad state!" << std::endl;
177 return -1;
178 }
179
180 if (m_zmq == nullptr) {
181 return -1;
182 }
183
184 if (m_type == Type::SUB) {
185 std::cout << "Error: Cannot send on SUB socket!" << std::endl;
186 return -1;
187 }
188
189 if (m_sendTransport != this) {
190 std::cout << "Error: This transport used for receive only!" << std::endl;
191 return -1;
192 }
193
194 // Create a local copy to modify the length
195 DmqHeader headerCopy = header;
196
197 // Calculate payload size and set it on the copy
198 std::string payload = os.str();
199 if (payload.length() > UINT16_MAX) {
200 std::cerr << "Error: Payload too large for 16-bit length." << std::endl;
201 return -1;
202 }
203 headerCopy.SetLength(static_cast<uint16_t>(payload.length()));
204
205 xostringstream ss(std::ios::in | std::ios::out | std::ios::binary);
206
207 // Write header values from the COPY
208 auto marker = htons(headerCopy.GetMarker());
209 ss.write(reinterpret_cast<const char*>(&marker), sizeof(marker));
210
211 auto id = htons(headerCopy.GetId());
212 ss.write(reinterpret_cast<const char*>(&id), sizeof(id));
213
214 auto seqNum = htons(headerCopy.GetSeqNum());
215 ss.write(reinterpret_cast<const char*>(&seqNum), sizeof(seqNum));
216
217 auto len = htons(headerCopy.GetLength());
218 ss.write(reinterpret_cast<const char*>(&len), sizeof(len));
219
220 // Insert delegate arguments (payload)
221 ss.write(payload.data(), payload.size());
222
223 size_t length = ss.str().length();
224
225 if (headerCopy.GetId() != dmq::ACK_REMOTE_ID)
226 {
227 if (m_transportMonitor)
228 m_transportMonitor->Add(headerCopy.GetSeqNum(), headerCopy.GetId());
229 }
230
231 // Send delegate argument data using ZeroMQ
232 int err = zmq_send(m_zmq, ss.str().c_str(), length, ZMQ_DONTWAIT);
233 if (err == -1)
234 {
235 // EAGAIN is common if queue is full, treat as error so RetryMonitor retries
236 return zmq_errno();
237 }
238 return 0;
239 }

241 virtual int Receive(xstringstream& is, DmqHeader& header) override
242 {
243 std::lock_guard<std::recursive_mutex> lock(m_mutex);
244
245 if (m_zmq == nullptr) {
246 // std::cout << "Error: Socket not created!" << std::endl;
247 return -1;
248 }
249
250 if (m_type == Type::PUB) {
251 std::cout << "Error: Cannot receive on PUB socket!" << std::endl;
252 return -1;
253 }
254
255 if (m_recvTransport != this) {
256 std::cout << "Error: This transport used for send only!" << std::endl;
257 return -1;
258 }
259
260 int size = zmq_recv(m_zmq, m_buffer, BUFFER_SIZE, 0);
261 if (size == -1) {
262 // Check if the error is due to a timeout
263 if (zmq_errno() == EAGAIN) {
264 //std::cout << "Receive timeout!" << std::endl;
265 }
266 return -1;
267 }
268
269 if (size < DmqHeader::HEADER_SIZE) {
270 std::cerr << "Received data is too small to process." << std::endl;
271 return -1;
272 }
273
274 xstringstream headerStream(std::ios::in | std::ios::out | std::ios::binary);
275
276 // Write the received header data into the stringstream
277 headerStream.write(m_buffer, DmqHeader::HEADER_SIZE);
278 headerStream.seekg(0);
279
280 uint16_t marker = 0;
281 headerStream.read(reinterpret_cast<char*>(&marker), sizeof(marker));
282 header.SetMarker(ntohs(marker));
283
284 if (header.GetMarker() != DmqHeader::MARKER) {
285 std::cerr << "Invalid sync marker!" << std::endl;
286 return -1; // @TODO: Optionally handle this case more gracefully
287 }
288
289 // Read the DelegateRemoteId (2 bytes) into the `id` variable
290 uint16_t id = 0;
291 headerStream.read(reinterpret_cast<char*>(&id), sizeof(id));
292 header.SetId(ntohs(id));
293
294 // Read seqNum using the getter for byte swapping
295 uint16_t seqNum = 0;
296 headerStream.read(reinterpret_cast<char*>(&seqNum), sizeof(seqNum));
297 header.SetSeqNum(ntohs(seqNum));
298
299 // Read length using the getter for byte swapping
300 uint16_t length = 0;
301 headerStream.read(reinterpret_cast<char*>(&length), sizeof(length));
302 header.SetLength(ntohs(length));
303
304 // Write the remaining target function argument data to stream
305 is.write(m_buffer + DmqHeader::HEADER_SIZE, size - DmqHeader::HEADER_SIZE);
306
307 if (header.GetId() == dmq::ACK_REMOTE_ID)
308 {
309 // Receiver ack'ed message. Remove sequence number from monitor.
310 if (m_transportMonitor)
311 m_transportMonitor->Remove(header.GetSeqNum());
312 }
313 else
314 {
315 if (m_transportMonitor && m_sendTransport)
316 {
317 // Send header with received seqNum as the ack message
318 xostringstream ss_ack;
319 DmqHeader ack;
320 ack.SetId(dmq::ACK_REMOTE_ID);
321 ack.SetSeqNum(header.GetSeqNum());
322
323 // Note: Recursive mutex allows us to call Send() if m_sendTransport == this.
324 // If m_sendTransport is a different instance, that instance's lock will be taken.
325 m_sendTransport->Send(ss_ack, ack);
326 }
327 }
328
329 return 0; // Success
330 }
