77 int Create(Type type, const char* addr_str, uint16_t port)
78 {
79 tx_mutex_get(&m_mutex, TX_WAIT_FOREVER);
80
81 m_type = type;
82 m_port = port;
83
84 // Parse IP Address
85 m_targetIp = 0;
86 if (addr_str) {
87 uint32_t a, b, c, d;
88 if (std::sscanf(addr_str, "%u.%u.%u.%u", &a, &b, &c, &d) == 4) {
89 // Construct IP ULONG (Assumes NetX expects Host Byte Order for setup)
90 m_targetIp = (a << 24) | (b << 16) | (c << 8) | d;
91 }
92 }
93
94 // Create the UDP Socket
95 UINT ret = nx_udp_socket_create(m_ip, &m_socket, (CHAR*)"DelegateMQ_Sock",
96 NX_IP_NORMAL, NX_FRAGMENT_OKAY, 0x80, 5);
97 if (ret != NX_SUCCESS) {
98 tx_mutex_put(&m_mutex);
99 return -1;
100 }
101
102 // Bind
103 if (type == Type::SUB) {
104 ret = nx_udp_socket_bind(&m_socket, port, NX_WAIT_FOREVER);
105 } else {
106 ret = nx_udp_socket_bind(&m_socket, NX_ANY_PORT, NX_WAIT_FOREVER);
107 }
108
109 if (ret != NX_SUCCESS) {
110 nx_udp_socket_delete(&m_socket);
111 tx_mutex_put(&m_mutex);
112 return -1;
113 }
114
115 tx_mutex_put(&m_mutex);

142 virtual int Send(xostringstream& os, const DmqHeader& header) override
143 {
144 tx_mutex_get(&m_mutex, TX_WAIT_FOREVER);
145
146 if (os.bad() || os.fail()) {
147 tx_mutex_put(&m_mutex);
148 return -1;
149 }
150
151 // Block regular data on SUB socket (only ACKs allowed)
152 if (m_type == Type::SUB && header.GetId() != dmq::ACK_REMOTE_ID) {
153 tx_mutex_put(&m_mutex);
154 return -1;
155 }
156
157 if (m_sendTransport != this) {
158 tx_mutex_put(&m_mutex);
159 return -1;
160 }
161
162 auto payload = os.str();
163 if (payload.length() > UINT16_MAX) {
164 tx_mutex_put(&m_mutex);
165 return -1;
166 }
167
168 // 1. Allocate NetX Packet
169 NX_PACKET* packet_ptr = nullptr;
170 UINT ret = nx_packet_allocate(m_pool, &packet_ptr, NX_UDP_PACKET, NX_NO_WAIT);
171 if (ret != NX_SUCCESS) {
172 tx_mutex_put(&m_mutex);
173 return -1;
174 }
175
176 // 2. Prepare Header (Network Byte Order)
177 DmqHeader headerCopy = header;
178 headerCopy.SetLength(static_cast<uint16_t>(payload.length()));
179
180 uint16_t marker = NETX_HTONS(headerCopy.GetMarker());
181 uint16_t id = NETX_HTONS(headerCopy.GetId());
182 uint16_t seqNum = NETX_HTONS(headerCopy.GetSeqNum());
183 uint16_t length = NETX_HTONS(headerCopy.GetLength());
184
185 // 3. Append Data (With Error Checks)
186 // Check return value of every append. If one fails, the packet is corrupted/partial.
187 if (nx_packet_data_append(packet_ptr, &marker, sizeof(marker), m_pool, NX_NO_WAIT) != NX_SUCCESS)
188 goto cleanup_and_fail;
189
190 if (nx_packet_data_append(packet_ptr, &id, sizeof(id), m_pool, NX_NO_WAIT) != NX_SUCCESS)
191 goto cleanup_and_fail;
192
193 if (nx_packet_data_append(packet_ptr, &seqNum, sizeof(seqNum), m_pool, NX_NO_WAIT) != NX_SUCCESS)
194 goto cleanup_and_fail;
195
196 if (nx_packet_data_append(packet_ptr, &length, sizeof(length), m_pool, NX_NO_WAIT) != NX_SUCCESS)
197 goto cleanup_and_fail;
198
199 if (nx_packet_data_append(packet_ptr, (VOID*)payload.data(), payload.size(), m_pool, NX_NO_WAIT) != NX_SUCCESS)
200 goto cleanup_and_fail;
201
202 // 4. Send
203 // nx_udp_socket_send typically releases the packet on internal errors.
204 // However, if it fails due to binding issues (before driver), it might not.
205 ret = nx_udp_socket_send(&m_socket, packet_ptr, m_targetIp, m_port);
206
207 if (ret != NX_SUCCESS) {
208 // Defensive release: Verify if your specific NetX version auto-releases on specific errors.
209 // Standard practice is often to release if send fails.
210 nx_packet_release(packet_ptr);
211 tx_mutex_put(&m_mutex);
212 return -1;
213 }
214
215 // 5. Track Reliability (after successful send only)
216 if (headerCopy.GetId() != dmq::ACK_REMOTE_ID && m_transportMonitor)
217 m_transportMonitor->Add(headerCopy.GetSeqNum(), headerCopy.GetId());
218
219 tx_mutex_put(&m_mutex);
220 return 0;
221
222 cleanup_and_fail:
223 nx_packet_release(packet_ptr);
224 tx_mutex_put(&m_mutex);
225 return -1;
226 }

228 virtual int Receive(xstringstream& is, DmqHeader& header) override
229 {
230 tx_mutex_get(&m_mutex, TX_WAIT_FOREVER);
231
232 if (m_recvTransport != this) {
233 tx_mutex_put(&m_mutex);
234 return -1;
235 }
236
237 NX_PACKET* packet_ptr = nullptr;
238
239 // Use configured timeout
240 UINT ret = nx_udp_socket_receive(&m_socket, &packet_ptr, m_recvTimeout);
241
242 if (ret != NX_SUCCESS) {
243 tx_mutex_put(&m_mutex);
244 return -1; // No packet or error
245 }
246
247 // --- Extract Source Info for ACKs ---
248 ULONG srcIp;
249 UINT srcPort;
250 nx_udp_source_extract(packet_ptr, &srcIp, &srcPort);
251
252 if (m_type == Type::SUB) {
253 m_targetIp = srcIp;
254 m_port = srcPort;
255 }
256
257 // --- Read Data ---
258 ULONG bytesRead;
259 UCHAR buffer[1500];
260
261 // Validate total length before attempting retrieve
262 if (packet_ptr->nx_packet_length > sizeof(buffer)) {
263 nx_packet_release(packet_ptr);
264 tx_mutex_put(&m_mutex);
265 return -1; // Packet too large for buffer
266 }
267
268 // Retrieve copies data out of packet (handling chained packets if necessary)
269 ret = nx_packet_data_retrieve(packet_ptr, buffer, &bytesRead);
270
271 nx_packet_release(packet_ptr);
272
273 if (ret != NX_SUCCESS || bytesRead < DmqHeader::HEADER_SIZE) {
274 tx_mutex_put(&m_mutex);
275 return -1;
276 }
277
278 // Parse Header
279 uint16_t* ptr16 = (uint16_t*)buffer;
280 header.SetMarker(NETX_NTOHS(ptr16[0]));
281 header.SetId(NETX_NTOHS(ptr16[1]));
282 header.SetSeqNum(NETX_NTOHS(ptr16[2]));
283 header.SetLength(NETX_NTOHS(ptr16[3]));
284
285 if (header.GetMarker() != DmqHeader::MARKER) {
286 tx_mutex_put(&m_mutex);
287 return -1;
288 }
289
290 // Check payload bounds
291 if (bytesRead < (ULONG)(DmqHeader::HEADER_SIZE + header.GetLength())) {
292 tx_mutex_put(&m_mutex);
293 return -1;
294 }
295
296 // Reset stream state before writing
297 is.clear();
298 is.str("");
299
300 // Write Payload to stream
301 is.write((char*)(buffer + DmqHeader::HEADER_SIZE), header.GetLength());
302
303 // --- ACK Logic ---
304 if (header.GetId() == dmq::ACK_REMOTE_ID) {
305 if (m_transportMonitor) m_transportMonitor->Remove(header.GetSeqNum());
306 }
307 else if (m_transportMonitor && m_sendTransport) {
308 // Send ACK
309 xostringstream ss_ack;
310 DmqHeader ack;
311 ack.SetId(dmq::ACK_REMOTE_ID);
312 ack.SetSeqNum(header.GetSeqNum());
313 ack.SetLength(0);
314
315 // Release lock before recursive call to Send to prevent deadlock
316 tx_mutex_put(&m_mutex);
317 m_sendTransport->Send(ss_ack, ack);
318 return 0;
319 }
320
321 tx_mutex_put(&m_mutex);
322 return 0;
323 }
