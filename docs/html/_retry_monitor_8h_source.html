74 int SendWithRetry(dmq::xostringstream& os, const dmq::transport::DmqHeader& header)
75 {
76 // Critical Section: Store the packet for retry before sending.
77 // If Send() fails we remove the entry immediately so it doesn't leak.
78 {
79 std::lock_guard<dmq::RecursiveMutex> lock(m_lock);
80 RetryEntry entry;
81 entry.attemptsRemaining = m_maxRetries;
82 entry.header = header;
83 entry.packetData = os.str(); // Copy data
84 m_retryStore[header.GetSeqNum()] = entry;
85 }
86
87 // Non-Critical Section: Send via Transport.
88 // We must NOT hold m_lock while calling Send().
89 // Send() calls TransportMonitor::Add(), which takes its own lock.
90 int result = m_transport.Send(os, header);
91
92 // If the send failed, TransportMonitor::Add() was never called so
93 // OnStatusChanged() will never fire for this seqNum. Remove the entry
94 // now to prevent it from leaking in m_retryStore indefinitely.
95 if (result != 0)
96 {
97 std::lock_guard<dmq::RecursiveMutex> lock(m_lock);
98 m_retryStore.erase(header.GetSeqNum());
99 }
100
101 return result;
102 }

104private:
106 void OnStatusChanged(dmq::DelegateRemoteId id, uint16_t seqNum, TransportMonitor::Status status)
107 {
108 (void)id;
109 // Variables to hold data for the retry OUTSIDE the lock
110 bool shouldRetry = false;
111 std::string retryPayload;
112 dmq::transport::DmqHeader retryHeader;
113
114 {
115 // 1. Critical Section: Read/Modify Map ONLY
116 const std::lock_guard<dmq::RecursiveMutex> lock(m_lock);
117
118 auto it = m_retryStore.find(seqNum);
119 if (it == m_retryStore.end()) return;
120
121 if (status == TransportMonitor::Status::SUCCESS)
122 {
123 // Message arrived safely, we can forget about the data now
124 m_retryStore.erase(it);
125 return; // Done
126 }
127 else if (status == TransportMonitor::Status::TIMEOUT)
128 {
129 if (it->second.attemptsRemaining > 0)
130 {
131 // Decrement counter
132 it->second.attemptsRemaining--;
133
134 // COPY data to local variables so we can use them after unlocking
135 retryPayload = it->second.packetData;
136 retryHeader = it->second.header;
137 shouldRetry = true;
138 }
139 else
140 {
141 // Max retries exceeded. Clean up.
142 // LOG_ERROR("RetryMonitor: Max retries exceeded for seq {}", seqNum);
143 m_retryStore.erase(it);
144 }
145 }
146 } // <--- LOCK IS RELEASED HERE
147
148 // 2. Non-Critical Section: Perform blocking network operations
149 if (shouldRetry)
150 {
151 // Re-prepare the stream from our local copy
152 dmq::xostringstream os(std::ios::in | std::ios::out | std::ios::binary);
153 os.write(retryPayload.data(), retryPayload.size());
154
155 // Re-send. The transport will re-add this to the TransportMonitor.
156 // This runs without holding m_lock, preventing a deadlock.
157 m_transport.Send(os, retryHeader);
158 }
159 }
