99 if (it == m_retryStore.end()) return;
100
101 if (status == TransportMonitor::Status::SUCCESS)
102 {
103 // Message arrived safely, we can forget about the data now
104 m_retryStore.erase(it);
105 return; // Done
106 }
107 else if (status == TransportMonitor::Status::TIMEOUT)
108 {
109 if (it->second.attemptsRemaining > 0)

112 it->second.attemptsRemaining--;

115 retryPayload = it->second.packetData;
116 retryHeader = it->second.header;
117 shouldRetry = true;
118 }
119 else

121 // Max retries exceeded. Clean up.
122 // LOG_ERROR("RetryMonitor: Max retries exceeded for seq {}", seqNum);
123 m_retryStore.erase(it);
124 }
125 }
126 } // <--- LOCK IS RELEASED HERE
127
128 // 2. Non-Critical Section: Perform blocking network operations

132 xostringstream os(std::ios::in | std::ios::out | std::ios::binary);
133 os.write(retryPayload.data(), retryPayload.size());
134
135 // Re-send. The transport will re-add this to the TransportMonitor.
136 // This runs without holding m_lock, preventing a deadlock.
